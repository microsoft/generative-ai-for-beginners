```python
        neg = random.choice(negative_examples)

        z = np.dot(pos, weights) # เดชเตเตผเดธเตเดชเตเดเตเดฐเตเตบ เดเดเตเดเตเดชเตเดเตเดเต เดเดฃเดเตเดเดพเดเตเดเตเด
        if z < 0: # เดชเตเดธเดฟเดฑเตเดฑเตเดตเต เดเดฆเดพเดนเดฐเดฃเด เดจเตเดเดฑเตเดฑเตเดตเดพเดฏเดฟ เดตเตผเดเตเดเตเดเดฐเดฟเดเตเดเต
            weights = weights + eta*weights.shape

        z  = np.dot(neg, weights)
        if z >= 0: # เดจเตเดเดฑเตเดฑเตเดตเต เดเดฆเดพเดนเดฐเดฃเด เดชเตเดธเดฟเดฑเตเดฑเตเดตเดพเดฏเดฟ เดตเตผเดเตเดเตเดเดฐเดฟเดเตเดเต
            weights = weights - eta*weights.shape

    return weights
```
