# មូលដ្ឋានវចនានុក្រមវិស្វកម្មបញ្ជា

## ការណែនាំ
មូឌុលនេះគ្របដណ្តប់ទៅលើយុទ្ធសាស្ត្រនិងបច្ចេកទេសសំខាន់ៗសម្រាប់បង្កើតបញ្ជាដែលមានប្រសិទ្ធភាពនៅក្នុងម៉ូឌែល AI បង្កើតថ្មី។ វិធីសាស្រ្តដែលអ្នកសរសេរបញ្ជាទៅ LLM ក៏មានសារៈសំខាន់ផងដែរ។ បញ្ជាដែលបានរៀបចំយ៉ាងប្រុងប្រយ័ត្នអាចសម្រេចបាននូវគុណភាពនៃការឆ្លើយតបល្អជាងមុន។ តែតើពាក្យដូចជា _បញ្ជា_ និង _វិស្វកម្មបញ្ជា_ មានន័យដូចម្ដេច? ហើយតើខ្ញុំធ្វើដូចម្តេចដើម្បីកែលម្អ _ការបញ្ចូលបញ្ជា_ ដែលខ្ញុំផ្ញើទៅ LLM? ពួកនេះជាសំនួរដែលយើងនឹងព្យាយាមឆ្លើយនៅក្នុងជំពូកនេះ និងជំពូកបន្ទាប់។

_ពូជ AI បង្កើតថ្មី_ អាចបង្កើតមាតិកាថ្មីៗ (ឧ. អត្ថបទ រូបភាព សំឡេង កូដ ទៀតទៀត) ដើម្បីឆ្លើយតបសំណើរបស់អ្នកប្រើប្រាស់។ វាធ្វើបានដោយប្រើ _ម៉ូឌែលភាសាធំ_ (Large Language Models) ដូចជា ស៊េរី GPT របស់ OpenAI ("Generative Pre-trained Transformer") ដែលត្រូវបានបណ្តុះបណ្តាលសម្រាប់ប្រើភាសាប្រពៃណី និងកូដ។

អ្នកប្រើប្រាស់អាចទំនាក់ទំនងជាមួយម៉ូឌែលទាំងនេះដោយប្រើបែបបទដែលស្គាល់គ្នាដូចជាការជជែកគ្នា ដោយមិនចាំបាច់មានជំនាញបច្ចេកទេស ឬការបណ្តុះបណ្តាល។ ម៉ូឌែលទាំងនេះគឺជា _ផ្អែកលើបញ្ជា_ - អ្នកប្រើប្រាស់ផ្ញើអត្ថបទបញ្ចូល (បញ្ជា) ហើយទទួលបានការឆ្លើយតបពី AI (ការសម្រេច) ត្រឡប់មកវិញ។ ហើយពួកគេចាប់ផ្តើម "ជជែកជាមួយ AI" ជាជួរដំណងមួយៗ ដើម្បីកែលម្អបញ្ជារបស់ខ្លួនរហូតដល់ការឆ្លើយតបស្មើតម្រូវការរបស់ពួកគេ។

“បញ្ជា” ឥឡូវនេះក្លាយជាចំណុចច្រកកម្មវិធីសំខាន់សម្រាប់កម្មវិធី AI បង្កើតថ្មី ដើម្បីប្រាប់ម៉ូឌែលអំពីអ្វីដែលត្រូវធ្វើ ហើយប៉ះពាល់គុណភាពនៃការឆ្លើយតបដែលបានប្រគល់វិញ។ "វិស្វកម្មបញ្ជា" គឺជាតំបន់សិក្សាដែលកំពុងរីកចម្រើនយ៉ាងរហ័ស ដែលផ្តោតលើការរចនា និងបង្កើតបញ្ជាឱ្យខ្ពស់ដើម្បីផ្តល់នូវការឆ្លើយតបត្រឹមត្រូវ និងមានគុណភាពជាស្រេច។

## គោលបំណងសិក្សា

នៅក្នុងមេរៀននេះ យើងសិក្សាអំពីអ្វីដែលជាវិស្វកម្មបញ្ជា មូលហេតុដែលវាសំខាន់ និងរបៀបដែលយើងអាចបង្កើតបញ្ជាដែលមានប្រសិទ្ធភាពសម្រាប់ម៉ូឌែល និងគោលបំណងកម្មវិធីជាក់លាក់មួយ។ យើងនឹងយល់ពីគំនិតមូលដ្ឋាន និងអនុវត្តន៍នូវបទប្បញ្ញត្តិល្អៗសម្រាប់វិស្វកម្មបញ្ជា - ហើយសម្រេចគោលបំណងក្នុងបរិយាកាស Jupyter Notebook "sandbox" ដែលអាចបង្ហាញឲ្យឃើញការអនុវត្តរបស់គំនិតទាំងនេះតាមអនុមានជាក់ស្តែង។

នៅចុងបក់នៃមេរៀននេះ យើងនឹងអាច៖

1. អធិប្បាយអំពីវិស្វកម្មបញ្ជា និងមូលហេតុដែលវាមានសារៈសំខាន់។
2. ពិពណ៌នាអំពីធាតុផ្សំរបស់បញ្ជា និងរបៀបប្រើប្រាស់វា។
3. រៀនអនុវត្តន៍បទប្បញ្ញត្តិល្អ និងបច្ចេកទេសសម្រាប់វិស្វកម្មបញ្ជា។
4. អនុវត្តន៍បច្ចេកទេសដែលបានរៀនទៅលើឧទាហរណ៍ពិត ជាមួយចំណុច OpenAI មួយ។

## ពាក្យគន្លឹះ

វិស្វកម្មបញ្ជា៖ ព្រឹត្តិការណ៍នៃការរចនា និងកែលម្អបញ្ជាទៅនាំម៉ូឌែល AI ទៅរកលទ្ធផលដែលចង់បាន។

Tokenization៖ ដំណើរការបំលែងអត្ថបទទៅជាឯកតារង់រាលតូចៗហៅថាតូកិន ដែលម៉ូឌែលអាចយល់ និងដំណើរការបាន។

Instruction-Tuned LLMs៖ ម៉ូឌែលភាសាធំដែលត្រូវបានបង្កើតបន្ថែមជាមួយនឹងការណែនាំជាក់លាក់ ដើម្បីកែលម្អភាពត្រឹមត្រូវនៃការឆ្លើយតប។

## បរិយាកាសសិក្សា Sandbox

វិស្វកម្មបញ្ជាគឺជាការប្រព្រឹត្តិមួយដែលបន្ថែមជាងវិទ្យាសាស្ត្រ។ វិធីល្អបំផុតក្នុងការកែលម្អនៃការយល់ដឹងគឺត្រូវ _អនុវត្តច្រើន_ និងអនុវត្តន៍វិធីសាស្ត្រ​ត្រួតពិនិត្យដោយការប្រើបញ្ហានៅក្នុងដែនកម្មវិធីជាមួយបច្ចេកទេសដែលផ្ដល់អនុសាសន៍ និងការបង្កើតម៉ូឌែលជាក់លាក់។

Jupyter Notebook ដែលភ្ជាប់ជាមួយមេរៀននេះផ្តល់បរិយាកាស _sandbox_ ដែលអ្នកអាចសាកល្បងអ្វីដែលបានរៀនបានតាមរយៈការចូលរួម ឬនៅផ្នែកនៃការប្រកួតកូដនៅចុងមេរៀន។ ដើម្បីអនុវត្តន៍លំហាត់ អ្នកត្រូវការប្រកួតខាងក្រោម៖

1. **គន្លឹះ API Azure OpenAI** - ចំណុចផ្ដល់សេវាសម្រាប់ LLM ដែលបានចាត់ចែង។
2. **បរិយាកាស Python Runtime** - ដែលអាចអនុវត្តបានលើ Notebook។
3. **ខួបបរិយាកាសក្នុងតំបន់** - _បញ្ចប់ជំហ៊ាន [SETUP](./../00-course-setup/02-setup-local.md?WT.mc_id=academic-105485-koreyst) ឥឡូវនេះសម្រាប់ការរៀបចំត្រៀមខ្លួន_។

Notebook មានលំហាត់ _ចាប់ផ្ដើម_ ល្អៗ ប៉ុន្តែអ្នកគួរតែបន្ថែម _Markdown_ (ការពិពណ៌នា) និង _Code_ (សំណើបញ្ជា) ដើម្បីសាកល្បងឧទាហរណ៍ឬគំនិតបន្ថែម និងពង្រឹងការយល់ដឹងរបស់អ្នកចំពោះការរចនាបញ្ជា។

## មគ្គុទិសបញ្ជាក់

ចង់ទទួលបានទិដ្ឋភាពធំទូលាយអំពីមេរៀននេះមុនពេលចាប់ផ្តើម? ពិនិត្យមើលមគ្គុទិសបញ្ជាក់ដែលបង្ហាញពីប្រធានបទសំខាន់ៗ និងចំណុចសំខាន់ដែលអាចគិតសម្រាប់មិត្តអ្នកអាននៅក្នុងមួយមុខទាំងអស់។ ផែនទីមេរៀននាំឲ្យអ្នកពីការយល់ដឹងអំពីកំណត់សំខាន់និងបញ្ហាផ្សេងៗ ដើម្បីដោះស្រាយដោយយកបច្ចេកទេសវិស្វកម្មបញ្ជានិងបទប្បញ្ញត្តិល្អៗមកអនុវត្ត។ សូមចំណាំថាផ្នែក "បច្ចេកទេសកម្រិតខ្ពស់" នៅក្នុងមគ្គុទិសនេះចំពោះមាតិកាដែលគ្របដណ្តប់នៅជំពូកក្រោយ។

## កម្មវិធីចាប់ផ្ដើមរបស់យើង

ឥឡូវនេះ មកពិភាក្សាអំពីរបៀបដែល _ប្រធានបទនេះ_ ទាក់ទងទៅនឹងបេសកកម្មស្តាតអាប់របស់យើងក្នុងការនាំឲ្យមានការបង្កើតប្រព័ន្ធ AI នៅក្នុងការអប់រំ [ប្រាប់ព័ត៌មានវិទ្យាជាមួយការសហការជាមួយ Microsoft](https://educationblog.microsoft.com/2023/06/collaborating-to-bring-ai-innovation-to-education?WT.mc_id=academic-105485-koreyst)។ យើងចង់បង្កើតកម្មវិធីដែលគាំទ្រដោយ AI សម្រាប់ _ការសិក្សាបុគ្គលិក_ - ដូច្នេះអាចគិតពីរបៀបដែលអ្នកប្រើប្រាស់ផ្សេងៗនៃកម្មវិធីយើងអាច "រចនា" បញ្ជា៖

- **អ្នកគ្រប់គ្រង** អាចស្នើឲ្យ AI _វិភាគទិន្នន័យគ្រប់គ្រងផែនការសិក្សាដើម្បីរកចំងល់ក្នុងការគ្របដណ្តប់_។ AI អាចសង្ខេបលទ្ធផលឬបង្ហាញវាដោយកូដ។
- **គ្រូបង្រៀន** អាចស្នើឲ្យ AI _បង្កើតផែនការបង្រៀនសម្រាប់គោលដៅនិងប្រធានបទមួយច្បាស់លាស់_។ AI អាចបង្កើតផែនការបុគ្គលិកជាមួយទ្រង់ទ្រាយបានកំណត់។
- **សិស្ស** អាចស្នើឲ្យ AI _បង្រៀនពួកគេជាមួយមុខវិជ្ជាពិបាកមួយ_។ AI អាចណែនាំសិស្សជាមួយមេរៀន ការបញ្ជាក់ និងឧទាហរណ៍ដែលសមនឹងកម្រិតរបស់ពួកគេ។

នេះគ្រាន់តែជាកំណត់ចំណុចតូចមួយប៉ុណ្ណោះ។ សូមពិនិត្យមើល [Prompts For Education](https://github.com/microsoft/prompts-for-edu/tree/main?WT.mc_id=academic-105485-koreyst) - បណ្ណាល័យបញ្ជាដែលមានកូដចំហទ្រព្យ ដែលបានត្រួតពិនិត្យដោយអ្នកជំនាញអប់រំ ដើម្បីយល់ពីអត្រាប្រយោជន៍ជាច្រើន! _សាកល្បងរត់បញ្ជាខ្លះនៅក្នុង sandbox ឬប្រើ OpenAI Playground ដើម្បីមើលអ្វីកើតឡើង!_

## តើអ្វីទៅជា វិស្វកម្មបញ្ជា?

យើងបានចាប់ផ្ដើមមេរៀននេះដោយកំណត់ **វិស្វកម្មបញ្ជា** ជាដំណើរការរចនា និងកែលម្អ _ការបញ្ចូលអត្ថបទ_ (បញ្ជា) ដើម្បីផ្តល់នូវការឆ្លើយតបដែលមានយុទ្ធសាស្ត្រ និងគុណភាពល្អសម្រាប់គោលបំណងកម្មវិធីនិងម៉ូឌែលជាក់លាក់មួយ។ យើងអាចគិតថាវាជាដំណើរការពីរជំហាន៖

- _រចនា_ បញ្ជាដំបូងសម្រាប់ម៉ូឌែលនិងគោលបំណងជាក់លាក់
- _កែលម្អ_ បញ្ជាជាបន្តបន្ទាប់ដើម្បីកែលម្អគុណភាពនៃការឆ្លើយតប

នេះជាការអនុវត្តបែបត្រៀមខ្លួន និងបញ្ហាសម្រាប់ការសាកល្បងដែលតម្រូវឲ្យមានអារម្មណ៍និងប្រឹងប្រែងរបស់អ្នកប្រើដើម្បីទទួលបានលទ្ធផលល្អបំផុត។ ហេតុអ្វីបានជាវាសំខាន់? ដើម្បីឆ្លើយសំនួរនេះ យើងត្រូវយល់ពីគំនិតបីរបៀប៖

- _Tokenization_ = របៀបដែលម៉ូឌែល "មើល" បញ្ជា
- _Base LLMs_ = របៀបដែលម៉ូឌែលមូលដ្ឋាន "ដំណើរការ" បញ្ជា
- _Instruction-Tuned LLMs_ = របៀបដែលម៉ូឌែលអាចត្រូវ "ការងារ"

### Tokenization

LLM មើលបញ្ជានៅជាសំណុំតូកិនដែលម៉ូឌែលផ្សេងៗ (ឬជំនាន់ម៉ូឌែលផ្សេងៗ) អាចបំបែកបញ្ជាដូចគ្នាតាមរបៀបខុសគ្នា។ ព្រោះ LLM ត្រូវបានបណ្តុះបណ្តាលលើតូកិន (មិនមែនអត្ថបទដើម) របៀបដែលបញ្ជាត្រូវបានបំបែកយ៉ាងខ្លាំងប៉ះពាល់ទៅលើគុណភាពនៃការឆ្លើយតបដែលបានបង្កើតឡើង។

ដើម្បីស្វែងយល់ពីរបៀបធ្វើការ Tokenization សូមសាកល្បងឧបករណ៍ដូចជា [OpenAI Tokenizer](https://platform.openai.com/tokenizer?WT.mc_id=academic-105485-koreyst) ដែលបង្ហាញខាងក្រោម។ ចម្លងបញ្ជារបស់អ្នកចូលក្នុងកន្លែងនេះ ហើយមើលថាវាត្រូវបានបំលែងជាតូកិនដូចម្តេច សូមផ្តោតសំខាន់លើរបៀបថា តួអក្សររឹង និងសញ្ញាពីសគឺត្រូវបានដោះស្រាយយ៉ាងដូចម្តេច។ សូមចំណាំថាឧទាហរណ៍នេះបង្ហាញ LLM ចាស់ (GPT-3) ដូច្នេះការសាកល្បងជាមួយម៉ូឌែលថ្មីអាចបង្កើតលទ្ធផលខុសប្លែក។

### គំនិត៖ ម៉ូឌែលមូលដ្ឋាន

បន្ទាប់ពីបញ្ជាត្រូវបានបំបែកជា Token បុគ្គលិកសំខាន់នៃ ["Base LLM"](https://blog.gopenai.com/an-introduction-to-base-and-instruction-tuned-large-language-models-8de102c785a6?WT.mc_id=academic-105485-koreyst) (ឬម៉ូឌែលមូលដ្ឋាន) គឺទាយរូបតូកិនបន្ទាប់ក្នុងរៀងខ្នាតនោះ។ ព្រោះ LLM ត្រូវបានបណ្តុះបណ្តាលលើសំណុំអត្ថបទធំទូលាយ ពួកគេមានការយល់ដឹងល្អអំពីទំនាក់ទំនងស្ថិតិរវាងតូកិន និងអាចទាយកូដនោះដោយមានទំនុកចិត្តមួយចំនួន។ សូមចំណាំថាពួកគេមិនយល់អំពី _អត្ថន័យ_ នៃពាក្យនៅក្នុងបញ្ជា ឬតូកិនទេ; ពួកគេគ្រាន់តែឃើញលំនាំមួយដែលពួកគេចង់"បញ្ចប់" ជាមួយការទាយកូដបន្ទាប់។ ពួកគេអាចបន្តទាយបន្តរហូតដល់មានការកាត់បន្ថយដោយអ្នកប្រើ ឬលក្ខខណ្ឌដែលបានកំណត់មុន។

ចង់បានមើលបែបការងារសម្រេចដែលផ្អែកលើបញ្ជា? សូមបញ្ចូលបញ្ជាខាងលើទៅក្នុង Azure OpenAI Studio [_Chat Playground_](https://oai.azure.com/playground?WT.mc_id=academic-105485-koreyst) ជាមួយការកំណត់លំនាំដើម។ ប្រព័ន្ធត្រូវបានកំណត់ឲ្យដឹកនាំបញ្ជាជាការស្នើសុំព័ត៌មាន ដូច្នេះអ្នកគួរតែឃើញសម្រេចដែលបំពេញតម្រូវការនេះ។

ប៉ុន្តែប្រសិនបើអ្នកប្រើចង់ឃើញអ្វីមួយជាក់លាក់ដែលសមរម្យតាមលក្ខខណ្ឌឬគោលបំណង? នេះជាកន្លែងដែល _Instruction-Tuned_ LLM ចូលមកជាបញ្ហា។

### គំនិត៖ Instruction Tuned LLMs

[Instruction Tuned LLM](https://blog.gopenai.com/an-introduction-to-base-and-instruction-tuned-large-language-models-8de102c785a6?WT.mc_id=academic-105485-koreyst) ចាប់ផ្ដើមពីម៉ូឌែលមូលដ្ឋាន ហើយសម្រួលបន្ថែមជាមួយឧទាហរណ៍ ឬគូបញ្ចូល/ចេញ (ឧ. "សារជាច្រើនដង") ដែលអាចមានការណែនាំច្បាស់ - ហើយការឆ្លើយតបពី AI ព្យាយាមតាមណែនាំនេះ។

វាប្រើបច្ចេកទេសដូចជា ការរៀនជាមួយមនុស្សប្រាក់ទុកមុខ (RLHF) ដែលអាចបណ្តុះបណ្តាលម៉ូឌែលឲ្យ _ធ្វើតាមការណែនាំ_ និង _រៀនពីមតិយោបល់_ ដើម្បីផលិតការឆ្លើយតបដែលសមរម្យសម្រាប់កម្មវិធីជាក់ស្តែង និងជាប់ទាក់ទងទៅនិងគោលបំណងអ្នកប្រើប្រាស់។

សូមសាកល្បងវិញ - ត្រឡប់ទៅបញ្ជាខាងលើ ប៉ុន្តែនូវផ្លាស់ប្តូរប្រែសារប្រព័ន្ធ _system message_ ដើម្បីផ្តល់ការណែនាំដូចតទៅជាបរិបទ៖

> _សង្ខេបមាតិកាដែលអ្នកបានផ្តល់សម្រាប់សិស្សថ្នាក់ទីពីរ។ រក្សាលទ្ធផលឲ្យនៅក្នុងអត្ថបទមួយដែលមានចំណុចបញ្ជា 3-5។_

មើលថាលទ្ធផលឥឡូវត្រូវបានលៃតម្រូតទៅនឹងគោលបំណង និងទ្រង់ទ្រាយដែលចង់បានយ៉ាងដូចម្តេច? គ្រូបង្រៀនអាចប្រើប្រាស់ការឆ្លើយតបនេះភ្លាមៗក្នុងស្លាយសម្រាប់ថ្នាក់។

## ហេតុអ្វីបានជាយើងត្រូវការ វិស្វកម្មបញ្ជា?

ឥឡូវនេះដែលយើងបានដឹងថាបញ្ជាត្រូវបានដំណើរការដោយ LLMs ដូចម្តេច អោយនិយាយអំពី _ហេតុអ្វី_ ប្រើវិស្វកម្មបញ្ជា។ ចម្លើយបង្កើតពីការពិតថា LLMs បច្ចុប្បន្នមានបញ្ហាផ្សេងៗដែលធ្វើឲ្យការសម្រេចត្រឹមត្រូវ រៀបរាប់ដែលអាចជាបញ្ហា ប្រសិនបើមិនមានការប្រាក់ប្រែការសរសេរបញ្ជា និងការកែលម្អទេ។ ឧទាហរណ៍៖

1. **ការឆ្លើយតបរបស់ម៉ូឌែលមានកម្រិតស៊ូតូស្តិក។** _បញ្ជាដូចគ្នា_ មិននៅតែបង្កើតការឆ្លើយតបដូចគ្នានៅជាមួយម៉ូឌែលផ្សេងៗ ឬជំនាន់ម៉ូឌែលផ្សេងៗទេ។ ហើយវានៅអាចបង្កើតលទ្ធផលខុសគ្នាតាមចំនួនដងប្រើជាមួយម៉ូឌែលដូចគ្នានៅពេលផ្សេងៗបានផងដែរ។ _បច្ចេកទេសវិស្វកម្មបញ្ជា អាចជួយគាត់បន្ថយការប្រែប្រួលទាំងនេះដោយផ្តល់ដែនកំណត់ល្អជាងមុន_។

1. **ម៉ូឌែលអាចច្របូកច្របល់ការឆ្លើយតប។** ម៉ូឌែលនេះបណ្តុះបណ្តាលជាមុនជាមួយ _ទិន្នន័យធំជាមួយកំណត់_ មានន័យថាពួកគេមិនមានចំណេះដឹងអំពីគំនិតក្រៅសមាសភាគតំណាងនៃការបណ្តុះបណ្តាលនោះទេ។ ជាមូលដ្ឋាន វាអាចបង្កើតការឆ្លើយតបដែលមិនត្រឹមត្រូវ មានកំណត់ត្រាមិនពិត ឬផ្ទុយគ្នាផ្ទាល់ទៅនឹងការពិតដែលបានគោរព។ _បច្ចេកទេសវិស្វកម្មបញ្ជា ជួយអ្នកប្រើស្វែងរក និងកាត់បន្ថយការបង្កើតច្របូកច្របល់ចំនួននេះ ដូចជាសួរអំពីឯកសារ ឬហេតុផលពី AI_។

1. **សមត្ថភាពម៉ូឌែលនឹងខុសគ្នា។** ម៉ូឌែលថ្មីឬជំនាន់ថ្មីនឹងមានសមត្ថភាពច្រើនជាង ប៉ុន្តែបាត់បង់លក្ខណៈពិសេស និងកំរៃនិងភាពស្មុគស្មាញ។ _វិស្វកម្មបញ្ជា អាចជួយយើងបង្កើតអនុវត្តន៍ល្អ និងការងារជាប្រព័ន្ធដែលអាចអនុវត្តបានដោយមិនគិតពីភាពខុសគ្នា ហើយអាចបង្រួមតម្រូវការម៉ូឌែលជាក់លាក់ដោយជាបន្តបន្ទាប់_។

សូមពិនិត្យមើលបែបនេះនៅលើ OpenAI ឬ Azure OpenAI Playground៖

- ប្រើបញ្ជាដូចគ្នាជាមួយការចាត់ចែង LLM ផ្សេងៗ (ជា OpenAI, Azure OpenAI, Hugging Face) - តើអ្នកឃើញការប្រែប្រួលទេ?
- ប្រើបញ្ជាដូចគ្នា ជាបន្តបន្ទាប់ជាមួយការចាត់ចែង LLM _ដូចគ្នា_ (ឧ. Azure OpenAI playground) - តើការប្រែប្រួលនេះខុសគ្នាយ៉ាងដូចម្តេច?

### ឧទាហរណ៍ការបង្កើតច្របូកច្របល់ (Fabrications)

នៅក្នុងវគ្គសិក្សានេះ យើងប្រើពាក្យ **"fabrication"** ដើម្បីយោងទៅលើហេតុការណ៍ដែល LLM ពេលខ្លះបង្កើតព័ត៌មានមិនត្រឹមត្រូវដោយសារពិបាកភាពក្នុងការបណ្តុះបណ្តាល ឬសៀវភៅឧបសគ្គផ្សេងៗ។ អ្នកប្រហែលបានឮពាក្យនេះក្នុងអត្ថបទ ឬការស្រាវជ្រាវដូចជា _"hallucinations"_។ ទោះជាយ៉ាងណា យើងណែនាំខ្លាំងឲ្យប្រើពាក្យ _"fabrication"_ ដើម្បីមិនធ្វើឲ្យវាមានទំលាប់មនុស្សដោយភ្ជាប់ពាក្យនេះជាគុណសម្បត្តិមនុស្សទៅលើលទ្ធផលដែលបង្កើតដោយម៉ាស៊ីនទេ។ វានៅសំខាន់សម្រាប់ការបញ្ជាក់នូវគោលការណ៍ [AI ទំនួលខុសត្រូវ](https://www.microsoft.com/ai/responsible-ai?WT.mc_id=academic-105485-koreyst) ពីមួយទស្សនៈពាក្យ ដើម្បីដកស្រង់ពាក្យដែលប្រហែលជាមានអារម្មណ៍ផ្ញើលើអ្នកឬមិនរួមបញ្ចូលក្នុងបរិបទខ្លះ។

ចង់បានយល់ពីរបៀបដែល fabrication ធ្វើការងារឫ? សូមគិតពីបញ្ជានៃការណែនាំ AI បង្កើតមាតិកាសម្រាប់ប្រធានបទមិនមាន (សម្រាប់ធានាថាវាមិនមាននៅក្នុងឃ្លាំងបណ្តុះបណ្តាលទេ)។ ឧទាហរណ៍ - ខ្ញុំបានសាកល្បងបញ្ជានេះ៖

> **បញ្ជា:** បង្កើតផែនការមេរៀនអំពីសង្រ្គាមក្នុងភពម៉ាស៊ីនឆ្នាំ 2076។
ការស្វែងរកវែបបានបង្ហាញឲ្យខ្ញុំឃើញថា មានតែកំណត់ហេតុខួបខ្មែរកំណត់ចិត្ត (ឧ. ស៊េរីទូរទស្សន៍ ឬសៀវភៅ) ស្ដីពីសម្រង់សង្គ្រាមលើផែនដីម៉ាតៀ - ប៉ុន្តែមិនមាននៅឆ្នាំ 2076 ទេ។ ចំណេះដឹងធម្មតាក៏ប្រាប់យើងថា ឆ្នាំ 2076 គឺជាពេលអនាគត ហើយដូច្នេះ មិនអាចត្រូវបានភ្ជាប់ជាមួយព្រឹត្តការណ៍ពិតបានឡើយ។

ដូចដែលបានជាភាពរំពឹង ទាំងម៉ូដែលនីមួយៗ (ឬកំណែម៉ូដែល) ផលិតចម្លើយខុសៗគ្នាខ្លះ ដោយសារប្រព័ន្ធភាគរយធម្មជាតិ និងភាពខុសគ្នាផ្នែកសមត្ថភាពម៉ូដែល។ ឧទាហរណ៍ ម៉ូដែលមួយមានគោលដៅសម្រាប់អ្នកសិស្សថ្នាក់ទី ៨ ខណៈដែលមួយផ្សេងទៀតកំពុងគិតថាគឺសិស្សវិទ្យាល័យ។ ប៉ុន្តែម៉ូដែលទាំងបីនេះបានបង្កើតចម្លើយដែលអាចធ្វើឲ្យអ្នកប្រើដែលមិនដឹងអ្វីប្រាកដថា ព្រឹត្តការណ៍នេះជាការពិត។

បច្ចេកទេសវិស្វកម្មសារ ដូចជា _metaprompting_ និង _ការកំណត់សីតុណ្ហភាព_ អាចកាត់បន្ថយការបង្កើតភាពក្លែងបន្លំបានខ្លះ។ រចនាសម្ព័ន្ធថ្មីនៃបច្ចេកទេសវិស្វកម្មសារក៏បញ្ចូលឧបករណ៍ និងបច្ចេកទេសថ្មីៗបានយ៉ាងទាន់ចិត្តក្នុងលំហូរសារដើម្បីបន្ថយឬកាត់បន្ថយផលប៉ះពាល់ទាំងនេះ។

## ការបង្កើតសារ

យើងបានឃើញហេតុផលថាហេតុអ្វីវិស្វកម្មសារមានសារៈសំខាន់ - ហើយឥលូវយើងនឹងយល់ពីរបៀបដែលសារត្រូវបាន _បង្កើត_ ដើម្បីយើងអាចវាយតម្លៃបច្ចេកទេសផ្សេងៗសម្រាប់រចនាសារដើម្បីមានប្រសិទ្ធភាពច្រើនជាងមុន។

### សារមូលដ្ឋាន

សូមចាប់ផ្តើមជាមួយសារមូលដ្ឋាន៖ អត្ថបទបញ្ចូលដែលផ្ញើទៅម៉ូដែលដោយគ្មានបរិបទផ្សេងទៀត។ ខាងក្រោមជាគំរូមួយ - នៅពេលយើងផ្ញើពាក្យម្ដងពីបទបទចម្រៀងជាតិអាមេរិកទៅ OpenAI [Completion API](https://platform.openai.com/docs/api-reference/completions?WT.mc_id=academic-105485-koreyst) វានៅតែ _បញ្ចប់_ ចម្លើយជាបន្ទាត់បន្ទាប់ទៀងទាត់បង្ហាញអាកប្បកិរិយាអនុគមន៍ប៉ារូ។

| សារ (បញ្ចូល)         | ការបញ្ចប់ (បញ្ចេញ)                                                                                                                                            |
| :--------------------- | :----------------------------------------------------------------------------------------------------------------------------------------- |
| Oh say can you see     | វាសម្លឹងដូចជាអ្នកកំពុងចាប់ផ្តើមបទចម្រៀង "The Star-Spangled Banner," ដែលជាបទចម្រៀងជាតិរបស់សហរដ្ឋអាមេរិក។ ខ្មែរមួយពេញវែងគឺ ... |

### សារលំបាក

ឥឡូវសូមបន្ថែមបរិបទនិងការណែនាំទៅលើសារមូលដ្ឋាននោះ។ [Chat Completion API](https://learn.microsoft.com/azure/ai-services/openai/how-to/chatgpt?WT.mc_id=academic-105485-koreyst) អនុញ្ញាតឲ្យយើងបង្កើតសារលំបាកជាការប្រមៀបសារជាច្រើនជាមួយៈ

- គូបញ្ចូល/បញ្ចេញដែលបង្ហាញពីការបញ្ចូលរបស់ _អ្នកប្រើ_ និងចម្លើយរបស់ _ជំនួយការ_។
- សារប្រព័ន្ធកំណត់បរិបទសម្រាប់អាកប្បកិរិយា ឬបុគ្គលិកលក្ខណៈរបស់ជំនួយការ។

សំណើគឺឥឡូវស្ថិតនៅក្នុងទ្រង់ទ្រាយខាងក្រោម ដែល _tokenization_ យកព័ត៌មាតិចំណុចសំខាន់ពីបរិបទនិងការពិភាក្សាចេញ។ ឥឡូវនេះ ការប្ដូរបរិបទប្រព័ន្ធអាចមានឥទ្ធិពលដូចគ្នានៅលើគុណភាពនៃការបញ្ចប់ ដូចជាការបញ្ចូលរបស់អ្នកប្រើបានផ្ដល់។

```python
response = openai.chat.completions.create(
    model="gpt-3.5-turbo",
    messages=[
        {"role": "system", "content": "You are a helpful assistant."},
        {"role": "user", "content": "Who won the world series in 2020?"},
        {"role": "assistant", "content": "The Los Angeles Dodgers won the World Series in 2020."},
        {"role": "user", "content": "Where was it played?"}
    ]
)
```

- **ឧទាហរណ៍** - ជំនួសបញ្ជាក់អ្វីដែលត្រូវធ្វើជាមួយការណែនាំត្រង់ៗ អ្នកផ្ដល់ឧទាហរណ៍អ្វីដែលត្រូវធ្វើ ហើយឲ្យម៉ូដែលរកគំរូ។
- **សញ្ញា** - តាមការណែនាំដោយមាន “សញ្ញា” ដែលរំញោចការបញ្ចប់ នាំម៉ូដែលឲ្យឆ្ពោះទៅចម្លើយដែលសមរម្យ។
- **ទំរង់** - គឺជាសម្រង់ដែលអាចធ្វើបម្រែបម្រួលបានសម្រាប់សារដែលមានជំនួសទីធៀប (អថេរ) ដែលអាចកំណត់តម្លៃដោយទិន្នន័យសម្រាប់ការប្រើប្រាស់ជាក់លាក់។

### រចនាសម្ព័ន្ធគំរូពាក្យស្នើរសុំ

គំរូពាក្យស្នើរសុំគឺជារៀបរាប់ដែលបានកំណត់ជាមុនសម្រាប់ពាក្យស្នើសុំមួយ ដែលអាចរក្សាទុក និងប្រើឡើងវិញបានពេលដែលត្រូវការ ដើម្បីជំរុញបទពិសោធន៍អ្នកប្រើឲ្យមានភាពស្វាគមន៍ជាប្រចាំ។ ក្នុងទ្រង់ទ្រាយសាមញ្ញបំផុត វាគឺជា ការប្រមូលផ្តុំនៃឧទាហរណ៍ពាក្យស្នើសុំ ដូចជា [ឧទាហរណ៍នេះពី OpenAI](https://cookbook.openai.com/examples/gpt4-1_prompting_guide?WT.mc_id=academic-105485-koreyst) ដែលផ្តល់មុខងារពាក្យស្នើសុំអន្តរកម្ម (សារ​អ្នកប្រើ និងប្រព័ន្ធ) និងទ្រង់ទ្រាយសំណើ API ដើម្បីគាំទ្រការប្រើឡើងវិញ។

ក្នុងទ្រង់ទ្រាយស្មុគស្មាញជាងនេះដូចជា [ឧទាហរណ៍ពី LangChain](https://python.langchain.com/docs/concepts/prompt_templates/?WT.mc_id=academic-105485-koreyst) វាមាន _ខ្ទង់ទទេ_ ដែលអាចជំនួសដោយទិន្នន័យពីប្រភពផ្សេងៗ (ការបញ្ចូលអ្នកប្រើ, សេចក្តីអនុសាសន៍ប្រព័ន្ធ, ទិន្នន័យខាងក្រៅ ជាដើម) ដើម្បីបង្កើតពាក្យស្នើសុំដោយស្វ័យប្រវត្តិ។ វាអនុញ្ញាតិឲ្យយើងបង្កើតបណ្ណាល័យពាក្យស្នើសូមដែលអាចប្រើឡើងវិញ ដើម្បីបង្កើតបទពិសោធន៍អ្នកប្រើបានយ៉ាងមានសន្ទុះ **ប្រព័ន្ធកម្មវិធី** ជារយៈកាលវែង។

ចុងក្រោយ តម្លៃពិតប្រាកដនៃគំរូសម្រាប់ពាក្យស្នើសុំ គឺស្ថិតនៅក្នុងសមត្ថភាពបង្កើត និងផ្សព្វផ្សាយ _បណ្ណាល័យពាក្យស្នើសុំ_ សម្រាប់ដែនកម្មវិធីបញ្ជំហើយ - ដែលគំរូពាក្យស្នើសុំត្រូវបាន _បង្កើតអោយសមរម្យ_ តាមបរិបទឬឧទាហរណ៍ដែលធ្វើឲ្យចម្លើយមានតំរូវការល្អ ឬត្រឹមត្រូវសម្រាប់អ្នកប្រើជាក់លាក់។ បណ្ណាល័យ [Prompts For Edu](https://github.com/microsoft/prompts-for-edu?WT.mc_id=academic-105485-koreyst) គឺជាឧទាហរណ៍ដ៏ល្អមួយក្នុងវិធីសាស្ត្រនេះ ដែលរើសយកបណ្ណាល័យនៃពាក្យស្នើសុំសម្រាប់ដែនការអប់រំជាមួយកំណត់លក្ខណៈសំខាន់ៗដូចជា ផែនការបង្រៀន ការរចនាផែនការ វគ្គសិក្សា ការបង្រៀនសិស្ស ហើយផ្សេងៗទៀត។

## ខ្លឹមសារគាំទ្រ

បើយើងគិតថាការបង្កើតពាក្យស្នើសុំមានដូចជា សេចក្តីណែនាំ (ភារកិច្ច) និង គោលដៅ (ខ្លឹមសារសំខាន់) តែ _ខ្លឹមសារថែម_ គឺជាបែបបទបន្ថែមដែលយើងផ្តល់ដើម្បី **ប៉ះពាល់ចំពោះលទ្ធផលមួយវិធីណាមួយ**។ វាអាចមានទ្រង់ទ្រាយបញ្ជា ការប្រមាណ ទ្រង់ទ្រាយ ការបែងចែកប្រធានបទ លក្ខណៈពិសេស ដែលជួយឲ្យម៉ូដែល _កំណត់លទ្ធផលឲ្យសមរម្យ_ នឹងគោលបំណងឬការរំពឹងទុករបស់អ្នកប្រើ។

ឧទាហរណ៍៖ ផ្គុំបញ្ជីវគ្គសិក្សាមួយដែលមានមាតិកាគ្រប់ស្ទុះ (ឈ្មោះ, ការពិពណ៌នា, កម្រិត, ស្លាហ្វីតមេតា, អ្នកបង្រៀន ជាដើម) ក្នុងគោលបំណងថា៖

- យើងអាចកំណត់សេចក្តីណែនាំ "សង្ខេបបញ្ជីវគ្គសិក្សាប្រចាំរដូវស Learn2023"
- យើងអាចប្រើខ្លឹមសារសំខាន់ឲ្យផ្តល់ឧទាហរណ៍ខ្លះៗនៃលទ្ធផលដែលចង់បាន
- យើងអាចប្រើខ្លឹមសារថែមដើម្បីកំណត់ ៥ ស្លាកដ៏រីកចម្រើនជាងគេ។

ឥឡូវនេះ ម៉ូដែលអាចផ្តល់សង្ខេបដោយទ្រង់ទ្រាយដែលបង្ហាញដោយឧទាហរណ៍ខ្លះៗ - ប៉ុន្តែប្រសិនបើលទ្ធផលមានស្លាកច្រើន វាអាចផ្តល់អាទិភាពដល់ ៥ ស្លាកដែលកំណត់ក្នុងខ្លឹមសារថែម។

## مដល់ពាក្យស្នើសុំប្រសើរបំផុត

ឥឡូវនេះដែលយើងដឹងពីរបៀបដែលពាក្យស្នើសុំអាច _ត្រូវបានចងក្រង_ យើងអាចចាប់ផ្តើមគិតពីរបៀប _រចនា_ ពួកវាឲ្យបង្ហាញពីបទពិសោធន៍ល្អបំផុត។ យើងអាចគិតពីវាផ្នែកពីរ៖ មាន _គំនិតចិត្តចុងក្រោយ_ ដែលត្រូវនិងអនុវត្ត _បច្ចេកទេស_ ត្រឹមត្រូវ។

### គំនិតចិត្តចុងក្រោយរបស់ការកំណត់ពាក្យស្នើសុំ

ការកំណត់ពាក្យស្នើសុំគឺជាដំណើរការប្រមាណ និងកែតម្រូវ ដូច្នេះសូមរក្សាទុកបីកត្តាធំៗជារៀងរាល់យ៉ាងក្នុងចិត្ត៖

1. **ការយល់ដឹងពីដែនកំណត់មានសារៈសំខាន់។** ភាពត្រឹមត្រូវនៃចម្លើយ និងភាពពាក់ព័ន្ធ គឺជាការងាររបស់ _ដែនកំណត់_ ដែលកម្មវិធីឬអ្នកប្រើបានដំណើរការនៅក្នុង។ ប្រើចំណេះដឹង និងជំនាញដែនកំណត់សម្រាប់ **ប្ដូរបច្ចេកទេស** ឲ្យបានល្អបន្ថែម។ ឧទាហរណ៍ កំណត់ _បុគ្គលិកលក្ខណៈប្រភេទដែនកំណត់_ ក្នុងពាក្យស្នើសុំប្រព័ន្ធរបស់អ្នក ឬប្រើ _គំរូដែលសមរម្យរាងដែនកំណត់_ នៅពាក្យស្នើសុំអ្នកប្រើ។ ផ្តល់ខ្លឹមសារថែមដែលបង្ហាញពីបរិបទដែនកំណត់ ឬប្រើ _សញ្ញានិងឧទាហរណ៍ដែនកំណត់_ ដើម្បីណែនាំម៉ូដែលទៅឲ្យប្រើប្រាស់ដោយស្គាល់។

2. **ការយល់ដឹងពីម៉ូដែលមានសារៈសំខាន់។** យើងដឹងថា ម៉ូដែលគឺមានលក្ខណៈស្តូខាស៊ីស្តិចតាមធម្មជាតិ។ ប៉ុន្តែការអនុវត្តម៉ូដែលអាចប្រែប្រួលទៅតាមទិន្នន័យបណ្តុះបណ្តាលគេប្រើ (ចំណេះដឹងបានបណ្ដុះបណ្ដាលជាមុន), សមត្ថភាពដែលគេផ្ដល់ (ឧទាហរណ៍ តាម API ឬ SDK), និងប្រភេទខ្លឹមសារដែលគេដំណើរការ (ឧទាហរណ៍ កូដ ភាពយន្ត អត្ថបទ)។ យល់ដឹងពីខ្លាំងខ្សោយរបស់ម៉ូដែលដែលអ្នកកំពុងប្រើ ហើយប្រើចំណេះដឹងនោះដើម្បី _អាទិភាពភារកិច្ច_ ឬបង្កើត _គំរូដែលប្ដូរតាមតំរូវការខ្លួនឯង_ ដែលបានបង្កើតល្អសម្រាប់សមត្ថភាពម៉ូដែល។

3. **ការកែតម្រូវ និងផ្ទៀងផ្ទាត់មានសារៈសំខាន់។** ម៉ូដែលកំពុងអភិវឌ្ឍយ៉ាងលឿន ហើយបច្ចេកទេសសម្រាប់ កំណត់ពាក្យស្នើសុំ ក៏ដូច្នេះដែរ។ ជាអ្នកជំនាញដែនកំណត់ អ្នកអាចមានបរិបទ ឬវិធានការផ្សេងទៀតសម្រាប់កម្មវិធីរបស់អ្នក ដែលមិនត្រូវពាក់ព័ន្ធនឹងសហគមន៍ទូលំទូលាយ។ ប្រើឧបករណ៍ និងបច្ចេកទេសកំណត់ពាក្យស្នើសុំ ដើម្បី "ចាប់ផ្តើម" ការបង្កើតពាក្យស្នើសុំ បន្ទាប់មកកែតម្រូវ និងផ្ទៀងផ្ទាត់លទ្ធផលដោយការយល់ដឹងផ្ទាល់ខ្លួន និងជំនាញដែនកំណត់។ កត់ត្រាចំណេះដឹងរបស់អ្នក និងបង្កើត **មូលដ្ឋានចំណេះ** (ឧទាហរណ៍ បណ្ណាល័យពាក្យស្នើសុំ) ដែលអាចប្រើជាកម្រិតដើមសម្រាប់អ្នកដទៃ សម្រាប់បញ្ចុះអត្រាកែតម្រូវលឿនឡើងនៅពេលខាងមុខ។
