# ការណែនាំអំពីបណ្តាញសរសៃប្រសាទ៖ Perceptron

ជាការព្យាយាមដំបូងមួយសម្រាប់អនុវត្តអ្វីដែលស្រដៀងទៅនឹងបណ្តាញសរសៃប្រសាទសម័យថ្មី ត្រូវបានធ្វើឡើងដោយ Frank Rosenblatt ពីបន្ទប់ពិសោធន៍ Cornell Aeronautical Laboratory ក្នុងឆ្នាំ 1957។ វាជាការអនុវត្តដោយរឹងហៅថា "Mark-1" ដែលបានរចនាឡើងដើម្បីស្គាល់រូបធម៌ចម្បងៗ ដូចជាមេញកែង បួនជ្រុង និងរង្វង់។

## ម៉ូដែល Perceptron

យើងសន្មតថាយើងមានគុណលក្ខណៈ N ក្នុងម៉ូដែលរបស់យើង ដែលនៅក្នុងករណីនេះវ៉ិចទ័របញ្ចូលនឹងមានទំហំពីរប្រាំពីរប្រាំពីរប្រាំពីរប្រាំពីរប្រភេទ N។ Perceptron គឺជាម៉ូដែល **ចំណាត់ថ្នាក់ពីរប្រភេទ** មានន័យថាវាអាចបំបែកចេញជាពីរប្រភេទនៃទិន្នន័យបញ្ចូល។ យើងនឹងសន្មតថាសម្រាប់វ៉ិចទ័របញ្ចូល x នីមួយៗ លទ្ធផលចេញពី perceptron របស់យើងនឹងមានតែ +1 ឬ -1 ដោយផ្អែកទៅលើប្រភេទ។ លទ្ធផលនឹងគណនាដោយប្រើរូបមន្ត៖

y(x) = f(w<sup>T</sup>x)

ដែល f គឺជាឧបករណ៍សកម្មភាពជំហាន

## ការបណ្តុះបណ្តាល Perceptron

ដើម្បីបណ្តុះបណ្តាល perceptron យើងត្រូវរកវ៉ិចទ័របំណង w ដែលចំណាត់ថ្នាក់បានត្រឹមត្រូវបំផុតសម្រាប់តម្លៃភាគច្រើន អ្វីដែលមានន័យថាបង្កើត **កំហុស** តិចបំផុត។ កំហុសនេះត្រូវបានកំណត់ដោយ **គោលការណ៍ perceptron** ដូច្នេះ៖

E(w) = -&sum;w<sup>T</sup>x<sub>i</sub>t<sub>i</sub>

ដោយដែលៈ

* គំនូសសរុបត្រូវបានយកលើចំណុចទិន្នន័យបណ្តុះបណ្តាល i ដែលបណ្តាលឱ្យមានការចំណាត់ថ្នាក់ខុស
* x<sub>i</sub> គឺទិន្នន័យបញ្ចូល និង t<sub>i</sub> គឺ -1 ឬ +1 សម្រាប់ឧទាហរណ៍អវិជ្ជមាន និងវិជ្ជមាន តាមលំដាប់។

គោលការណ៍នេះត្រូវបានពិចារណាថាជាឧបករណ៍មួយនៃវ៉ិចទ័រពន្យល់ w ហើយយើងត្រូវតែបន្ថយវា។ ជាញឹកញាប់វិធីសាស្រ្តមួយហៅថា **gradient descent** ត្រូវបានប្រើ ដែលយើងចាប់ផ្ដើមពីវ៉ិចទ័រពន្យល់ដើម w<sup>(0)</sup> ហើយនៅកម្រិតនីមួយៗធ្វើការកែប្រែវ៉ិចទ័រតាមរូបមន្ត៖

w<sup>(t+1)</sup> = w<sup>(t)</sup> - &eta;&nabla;E(w)

នៅទីនេះ &eta; គឺជាគន្លងហៅថា **អត្រាសិក្សា** និង &nabla;E(w) បង្ហាញពី **gradient** របស់ E។ បន្ទាប់ពីគណនាយោង gradient យើងទទួលបាន

w<sup>(t+1)</sup> = w<sup>(t)</sup> + &sum;&eta;x<sub>i</sub>t<sub>i</sub>

Algorithm ក្នុង Python មានរូបរាងដូចខាងក្រោម៖

```python
def train(positive_examples, negative_examples, num_iterations = 100, eta = 1):

    weights = [0,0,0] # ចាប់ផ្តើមគណនត្ថភាគទំងន់ (ជិតៗគ្នាបែបចៃដន្យ :)
        
    for i in range(num_iterations):
        pos = random.choice(positive_examples)
        neg = random.choice(negative_examples)

        z = np.dot(pos, weights) # គណនលទ្ធផលបហិទិទ័យ
        if z < 0: # ឧទាហរណ៍វិជ្ជមានត្រូវបានចាត់ថ្នាក់ជាវិជ្ជមាន
            weights = weights + eta*weights.shape

        z  = np.dot(neg, weights)
        if z >= 0: # ឧទាហរណ៍អវិជ្ជមានត្រូវបានចាត់ថ្នាក់ជាអវិជ្ជមាន
            weights = weights - eta*weights.shape

    return weights
```
