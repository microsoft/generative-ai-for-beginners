# ស៊ុមបណ្ដាញប្រព័ន្ធប្រតិបត្តិការ

ដូចដែលយើងបានរៀនរួចមកហើយ ដើម្បីអាចបណ្តុះបណ្តាលស៊ុមបណ្ដាញប្រព័ន្ធប្រតិបត្តិការ (neural networks) បានយ៉ាងមានប្រសិទ្ធភាព យើងត្រូវធ្វើរឿងពីរដូចខាងក្រោម៖

* ដើម្បីដំណើរការលើតែនស័រ ឧទាហរណ៍ ដើម្បីគុណ បូក និងគណនាអនុគមន៍ខ្លះៗដូចជា sigmoid ឬ softmax
* ដើម្បីគណនាឯកសណ្ឋាន gradient នៃការបង្ហាញទាំងអស់ ដើម្បីអនុវត្តបច្ចេកទេស gradient descent ដើម្បីបង្កើតយ៉ាងមានប្រសិទ្ធភាព



ខណៈពេលដែលបណ្ណាល័យ `numpy` អាចធ្វើការលើផ្នែកទីមួយនេះបាន ប៉ុន្តែយើងត្រូវការបច្ចេកទេសខ្លះណាមួយក្នុងការគណនាឯកសណ្ឋាន gradient។ ក្នុងស៊ុមបណ្ដាញដែលយើងបានអភិវឌ្ឍនៅក្នុងផ្នែកមុន យើងត្រូវតែកម្មវិធីដៃសម្រាប់អនុគមន៍បូកលំបាកទាំងអស់នៅក្នុងវិធីសាស្រ្ត `backward` ដែលធ្វើបច្ចេកទេស backpropagation។ ក្នុងលក្ខខណ្ឌល្អបំផុត ស៊ុមបណ្ដាញគួរតែផ្តល់ឱកាសឱ្យយើងគណនាឯកសណ្ឋាន gradient នៃ *រូបមន្តណាមួយ* ដែលយើងអាចកំណត់បាន។

អ្វីមួយសំខាន់បន្ថែមទៀតគឺត្រូវអាចធ្វើការគណនាលើ GPU ឬឧបករណ៍គណនាពិសេសផ្សេងទៀត ដូចជា TPU។ ការបណ្តុះបណ្តាលស៊ុមបណ្ដាញប្រព័ន្ធប្រតិបត្តិការ (deep neural network) ត្រូវការការគណនាច្រើនណាស់ ហើយអាចអនុវត្តនៅលើ GPUs ដើម្បីធ្វើការការបែងចែកគណនាឲ្យទៅតាមឧបករណ៍ជាច្រើនគឺមានសារៈសំខាន់ខ្លាំង។

> ✅ ពាក្យ 'parallelize' មានន័យថាការចែកចាយការគណនាទៅឧបករណ៍ជាច្រើន។

បច្ចុប្បន្ននេះ ស៊ុមបណ្ដាញប្រព័ន្ធប្រតិបត្តិការ ២ ប្រភេទដែលពេញនិយមបំផុតគឺ TensorFlow និង PyTorch។ ទាំងពីរមាន API កម្រិតទាបសម្រាប់ប្រតិបត្តិការជាមួយតែនស័រលើ CPU និង GPU។ លើស API កម្រិតទាប មានកម្រិត API ខ្ពស់ជាង ដែលមានឈ្មោះថា Keras និង PyTorch Lightning តាមលំដាប់។

Low-Level API | TensorFlow| PyTorch
--------------|-------------------------------------|--------------------------------
High-level API| Keras| Pytorch

**API កម្រិតទាប** ក្នុងទាំងពីរស៊ុមបណ្ដាញអនុញ្ញាតឲ្យអ្នកបង្កើតអ្វីដែលហៅថា **graft គណនាកម្មវិធី** (computational graphs)។ ក្រាហ្វនេះកំណត់នូវរបៀបគណនាចេញលទ្ធផល (ជាទូទៅគឺ loss function) ជាមួយប៉ារ៉ាម៉ែត្រ​ដែលមាន ហើយអាចផ្ញើរ​រត់​នៅលើ GPU ប្រសិនបើមាន។ មានអនុគមន៍សម្រាប់បំបែកក្រាហ្វគណនានេះ និងគណនាឯកសណ្ឋាន gradient ដែលបន្ទាប់មកអាចប្រើសម្រាប់ប្រសិទ្ធភាពប៉ារ៉ាម៉ែត្រ​ម៉ូដែល។

**API កម្រិតខ្ពស់** ជាទូទៅ គិតស៊ុមបណ្ដាញប្រព័ន្ធប្រតិបត្តិការ​ជា **ជួរដេរផ្នែក** (sequence of layers) ហើយធ្វើឲ្យការបង្កើតស៊ុមបណ្ដាញភាគច្រើនកាន់តែងាយស្រួល។ ការបណ្តុះបណ្តាលម៉ូដែលជាទូទៅត្រូវការការរៀបចំទិន្នន័យ ហើយបន្ទាប់មកហៅមុខងារ `fit` ដើម្បីធ្វើការងារ។

API កម្រិតខ្ពស់នេះអនុញ្ញាតឲ្យអ្នកបង្កើតស៊ុមបណ្ដាញប្រព័ន្ធប្រតិបត្តិការ​ធម្មតា​បានយ៉ាងលឿនដោយមិនចាំបាច់បារម្ភអំពីការពណ៌នារបស់វាលម្អិតច្រើនណា។ នៅពេលដដែល API កម្រិតទាបផ្តល់ការត្រួតពិនិត្យច្រើនជាងលើដំណើរការបណ្តុះបណ្តាល ហើយហេតុនេះវាត្រូវបានប្រើច្រើននៅក្នុងការស្រាវជ្រាវ លើស្ថាបត្យកម្មស៊ុមបណ្ដាញថ្មីៗ។

វាក៏មានសារៈសំខាន់ក្នុងការយល់ថាអ្នកអាចប្រើ API ទាំងពីរជាមួយគ្នាបាន។ ឧទាហរណ៍ អ្នកអាចអភិវឌ្ឍស្ថាបត្យកម្មគ្រឿងបន្ទាត់បណ្ដាញផ្ទាល់ខ្លួនដោយប្រើ API កម្រិតទាប ហើយបន្ទាប់មកប្រើវានៅក្នុងបណ្ដាញធំដែលបានបង្កើតនិងបណ្តុះបណ្តាលដោយ API កម្រិតខ្ពស់។ ឬអ្នកអាចកំណត់បណ្ដាញប្រព័ន្ធប្រតិបត្តិការ ដោយប្រើ API កម្រិតខ្ពស់ជាជួរដេរផ្នែក ហើយបន្ទាប់មកប្រើរង្វង់បណ្តុះបណ្តាលកម្រិតទាបផ្ទាល់ខ្លួន ដើម្បីអនុវត្តបច្ចេកទេស optimization។ API ទាំងពីរប្រើមូលដ្ឋានគំនិតដូចគ្នា និងរចនាឡើងដើម្បីធ្វើការសហការល្អជាមួយគ្នា។

## ការរៀន

ក្នុងវគ្គនេះ យើងផ្តល់មេរៀនភាគច្រើនសម្រាប់ PyTorch និង TensorFlow។ អ្នកអាចជ្រើសរើសស៊ុមការងារដែលអ្នកពេញចិត្ត ហើយតែអនុវត្តតាមសៀវភៅកំណត់គ្រប់គ្រាន់មួយ។ ប្រសិនបើអ្នកមិនប្រាកដថាតើស៊ុមណាដែលត្រូវជ្រើសរើស អ្នកអាចអានពិភាក្សាអំពី **PyTorch ប្រៀបធៀប TensorFlow** នៅលើអ៊ីនធឺណិត។ អ្នកក៏អាចមើលស៊ុមទាំងពីរដើម្បីយល់ដឹងជ្រៅជាង។

នៅកន្លែងដែលអាចធ្វើទៅបាន យើងនឹងប្រើ API កម្រិតខ្ពស់សម្រាប់ភាពងាយស្រួល។ ទោះជាយ៉ាងណា យើងជឿថាវាសំខាន់ក្នុងការយល់ដឹងពីរបៀបដែលស៊ុមបណ្ដាញប្រព័ន្ធប្រតិបត្តិការ ធ្វើការងារពីមូលដ្ឋានដំបូង ដូច្នេះនៅដើម យើងចាប់ផ្ដើមដោយប្រើ API កម្រិតទាប និងតែនស័រ។ ប៉ុន្តែ ប្រសិនបើអ្នកចង់ចាប់ផ្ដើមរហ័ស និងមិនចង់ចំណាយពេលវេលាច្រើនក្នុងការរៀនព័ត៌មានលម្អិតទាំងនេះ អ្នកអាចមិនបានរៀនផងដែរ ហើយទៅកាន់សៀវភៅកំណត់ API កម្រិតខ្ពស់យ៉ាងត្រង់។

# Overfitting

Overfitting គឺជាគំនិតមានសារៈសំខាន់ខ្លាំងក្នុងការសិក្សាបណ្ដាញបរិក្ខារមេកានិក ហើយវាសំខាន់ណាស់ក្នុងការប្រាកដថាអ្នកយល់ត្រឹមត្រូវ!

* នៅផ្នែកឆ្វេង យើងឃើញការប៉ាន់ប្រមាណការបង្ហាញបន្ទាត់ត្រឹមត្រូវល្អ។ ព្រោះចំនួនប៉ារ៉ាម៉ែត្រ​គឺសមរម្យ ម៉ូដែលយល់ដឹងនូវនៅពីក្រោយចំណុចផ្សព្វផ្សាយបានត្រឹមត្រូវ។
* នៅផ្នែកស្តាំ ម៉ូដែលកាន់តែមានសមត្ថភាពខ្លាំង។ ព្រោះយើងមានតែ ៥ ចំណុច ហើយម៉ូដែលមាន ៧ ប៉ារ៉ាម៉ែត្រ វាអាចចូលរួមកំណត់តែប៉ុណ្ណោះដើម្បីឆ្លងកាត់ចំណុចទាំងអស់ ដូច្នេះកំហុសបណ្តុះបណ្តាលកន្លែងចាប់បាន 0។ ទោះបីបែរតែនេះ វាបញ្ឈប់ម៉ូដែលពីការយល់ដឹងលើលំនាំត្រឹមត្រូវនៅពីក្រោយទិន្នន័យ ធ្វើឲ្យកំហុសផ្ទៀងផ្ទាត់ខ្ពស់ណាស់។

## របៀបស្គាល់ overfitting

ដូចដែលអ្នកឃើញពីក្រាហ្វខាងលើ overfitting អាចស្គាល់បានពីកំហុសបណ្តុះបណ្តាលទាបខ្លាំង និងកំហុសផ្ទៀងផ្ទាត់ខ្ពស់។ ជាទូទៅក្នុងដំណើរការបណ្តុះបណ្តាល យើងនឹងឃើញកំហុសទាំងពីរចាប់ផ្តើមកើនឡើងហើយបន្ថយចុះ បន្ទាប់ពីពេលខ្លះ កំហុសផ្ទៀងផ្ទាត់អាចរង់ចាំបញ្ឈប់បន្ថយ ហើយចាប់ផ្តើមកើនឡើងវិញ។ នេះគឺជាសញ្ញានៃ overfitting ហើយជាអ្នកបង្ហាញថាយើងគួរលែងបណ្តុះបណ្តាលនៅពេលនោះ (ឬយ៉ាងហោចណាស់ធ្វើក្រោយបង្ហាញម៉ូដែល)។

## របៀបការពារការលើសហ្វីទធីង

បើអ្នកមើលឃើញថា overfitting កើតមាន អ្នកអាចធ្វើលើកទីមួយ​ពីរបៀបដូចខាងក្រោម៖

 * បង្កើនបរិមាណទិន្នន័យបណ្តុះបណ្តាល
 * បន្ថយភាពស្មុគស្មាញរបស់ម៉ូដែល
 * ប្រើបច្ចេកទេស regularization មួយចំនួន ដូចជា Dropout ដែលយើងនឹងពិចារណាពេលក្រោយ។

## Overfitting និងការជួញដូរ Bias-Variance

overfitting ជាករណីជាក់លាក់មួយនៃបញ្ហាប្រូទិកក្នុងគណិតសាស្រ្តដែលហៅថា Bias-Variance Tradeoff។ ប្រសិនបើយើងពិចារណាតាមផ្លូវប្រភពកំហុសក្នុងម៉ូដែល យើងអាចឃើញកំហុសពីប្រភេទពីរ៖

* **កំហុស Bias** មកពីអាល់ហ្គរីធម៍របស់យើងមិនអាចចាប់យកទំនាក់ទំនងរវាងទិន្នន័យបណ្តុះបណ្តាលបានត្រឹមត្រូវ។ វាអាចកើតមកពីករណីដែលម៉ូដែលមិនមានសមត្ថភាពគ្រប់គ្រាន់ (**underfitting**)។
* **កំហុស Variance** ដែលមកពីម៉ូដែលចង់ប៉ាន់ប្រមាណសំលេងរំខានក្នុងទិន្នន័យបញ្ចូល មិនមែនទំនាក់ទំនងពិតប្រាកដដូចជា (**overfitting**)។

ក្នុងដំណើរការបណ្តុះបណ្តាល កំហុស bias នឹងកើនចុះ (ដោយសារតែម៉ូដែលរៀនកំណត់ទិន្នន័យ) ខណៈដែលកំហុស variance កើនឡើង។ វាសំខាន់ក្នុងការរលត់ការបណ្តុះបណ្តាល - ឬដោយដៃ (ពេលចាប់បាន overfitting) ឬដោយក្ដីស្វ័យប្រវត្តិ (ដោយបន្ថែម regularization) - ដើម្បីការពារការលើសហ្វីទធីង។
