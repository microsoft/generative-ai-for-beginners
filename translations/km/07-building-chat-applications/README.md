# ការសាងសង់កម្មវិធីជជែកដែលបង្កើតដោយ Generative AI

ពេលនេះ ដែលយើងបានឃើញរបៀបដែលយើងអាចសាងសង់កម្មវិធីបង្កើតអត្ថបទ កន្លែងនេះយើងនឹងស្វែងយល់អំពីកម្មវិធីជជែក។

កម្មវិធីជជែកបានវេចខ្ចប់ជារួមក្នុងជីវិតប្រចាំថ្ងៃរបស់យើង ដែលផ្តល់ជាងតែការសន្ទនាប្រកបដោយភាពរំភើបតេីវាជាផ្នែកសំខាន់នៃសេវាកម្មប្រកបដោយអតិថិជន ការគាំទ្របច្ចេកទេស និងរបៀបប្រកបដោយការប្រឹក្សាកម្រិតខ្ពស់។ អ្នកប្រហែលបានទទួលជំនួយពីកម្មវិធីជជែកមួយមិនយូរប៉ុន្មាន។ នៅពេលយើងបញ្ចូលបច្ចេកវិទ្យាទទួលស្គាល់ដូចជា generative AI ទៅក្នុងឆាកកីឡានេះ ការលំបាកនិងកើនឡើងដូចជាពិបាកផងដែរ។

សំណួរមួយចំនួនដែលយើងត្រូវបានឆ្លើយមានដូចជា៖

- **ការសាងសង់កម្មវិធី**। តើធ្វើដូចម្តេចដើម្បីសាងសង់ និងបញ្ចូលកម្មវិធីដែលមាន Generative AI បានយ៉ាងមានប្រសិទ្ធភាពសម្រាប់ករណីប្រើប្រាស់ជាក់លាក់?
- **ការត្រួតពិនិត្យ**។ បន្ទាប់ពីបានចេញផ្សាយ តើយើងគួរតែត្រួតពិនិត្យ និងធានាថាកម្មវិធីដំណើរការដោយគុណភាពខ្ពស់បំផុតក្នុងផ្នែកមុខងារ និងនៅតាម [មូលដ្ឋានទាំងប្រាំមួយនៃ AI ដែលទទួលខុសត្រូវ](https://www.microsoft.com/ai/responsible-ai?WT.mc_id=academic-105485-koreyst) อย่างไร?

នៅពេលដែលយើងចូលទៅក្នុងសម័យដែលបានកំណត់ដោយស្វ័យប្រវត្តិ និងការប្រតិបត្តិការប្រាក់មនុស្ស-ម៉ាស៊ីនដោយរលូន ការយល់ដឹងអំពីរបៀបដែល generative AI បម្លែងវិសាលភាព ជម្រៅ និងការប្រែប្រួលនៃកម្មវិធីជជែកក្លាយជារឿងសំខាន់។ មេរៀននេះនឹងស្រាវជ្រាវលើមុខងារពីរចនាសម្ព័ន្ធដែលគាំទ្រតួរនេះ ជ្រេីសដល់វិធីសាស្រ្តសម្រាប់កែតម្រូវវាជាក់លាក់ទៅលើបេសកកម្មដែន, និងវាយតម្លៃវិមាត្រនិងការបាំងបារម្ភដែលពាក់ព័ន្ធក្នុងការដាក់បញ្ចូល AI ដោយទទួលខុសត្រូវ។

## ការណែនាំ

មេរៀននេះគ្របដណ្តប់៖

- ជំនាញសម្រាប់សាងសង់ និងបញ្ចូលកម្មវិធីជជែកដោយប្រសិទ្ធភាព។
- របៀបអនុវត្តន៍ការប្ដូរតម្រូវ និងការកែតម្រូវយ៉ាងខ្លីទៅលើកម្មវិធី។
- គោលនយោបាយ និងការបង្កើតខ្លឹមសារដើម្បីត្រួតពិនិត្យកម្មវិធីជជែកយ៉ាងមានប្រសិទ្ធភាព។

## គោលបំណងសិក្សា

នៅចុងបញ្ចប់មេរៀននេះ អ្នកនឹងអាច៖

- ពណ៌នាអំពីការបង្កើត និងបញ្ចូលកម្មវិធីជជែកទៅក្នុងប្រព័ន្ធដែលមានស្រាប់។
- កែរប្រែកម្មវិធីជជែកសម្រាប់ករណីប្រើប្រាស់ជាក់លាក់។
- រកឃើញវិមាត្រសំខាន់ និងការពិចារណាគ្រឹះសម្រាប់ត្រួតពិនិត្យ និងថែរក្សាគុណភាពនៃកម្មវិធីជាAI-បង្កើតជជែក។
- ធានាថាកម្មវិធីជជែកបានប្រើ AI ដោយទទួលខុសត្រូវ។

## ការបញ្ចូល Generative AI ទៅកាន់កម្មវិធីជជែក

ការឡើងកម្រិតកម្មវិធីជជែកតាមរយៈ generative AI មិនមែនគ្រាន់តែធ្វើឲ្យវាឆ្លាតវៃប៉ុណ្ណោះទេ។ វាជាការប៉ាន់ប្រមាណរចនាសម្ព័ន្ធ, ប្រសិទ្ធភាព និងចំណុចប្រទាក់អ្នកប្រើដើម្បីផ្តល់បទពិសោធន៍ដែលមានគុណភាពល្អ។ វាគ្របដណ្តប់ពីមូលដ្ឋានរចនាសម្ព័ន្ធ, ការបញ្ចូល API និងការពិនិត្យចំណុចប្រទាក់អ្នកប្រើ។ ផ្នែកនេះមានគោលបំណងផ្តល់មាគ៌ាច្បាស់លាស់សម្រាប់អ្នកដើម្បីរៀបចំផ្លូវដំណើរចេញពីតំបន់ស្មុគស្មាញទាំងនេះ មិនថាអ្នកបញ្ចូលវាទៅក្នុងប្រព័ន្ធដែលមានស្រាប់ឬកំពុងសាងសង់វាជាប្លាតផ្លាតឯកតា។

នៅចុងផ្នែកនេះ អ្នកនឹងបានបំពាក់ជំនាញដែលចាំបាច់សម្រាប់សាងសង់ និងបញ្ចូលកម្មវិធីជជែកដោយប្រសិទ្ធភាព។

### ចូរយល់ពី chatbot ឬកម្មវិធីជជែក?

មុនពេលយើងចូលទៅក្នុងការសាងសង់កម្មវិធីជជែក សូមប្រៀបធៀបរវាង 'chatbots' និង 'កម្មវិធីជជែកដែលបានបង្កើតដោយ AI' ដែលមានតួនាទី និងមុខងារផ្សេងៗគ្នា។ បំណងដ៏សំខាន់នៃ chatbot គឺដំណើរការ​សកម្មភាពសន្ទនា​ជាក់លាក់ ដូចជាចម្លើយសំណួរញឹកញាប់ ឬតាមដានការផ្ញើសារវត្ថុ។ ជាធម្មតា វាត្រូវគ្រប់គ្រងដោយตรรก្យាសមាសនិងកូដ AI ស្មុគស្មាញ។ ខណៈដែលកម្មវិធីជជែកដែលមាន AI គឺជាបរិបទធំជាងនេះ ដែលផ្តល់នូវការទំនាក់ទំនងប្រភេទជាច្រើន ដូចជាចុះខ្លឹមសារ សំលេង និងវីដេអូរវាងអ្នកប្រើប្រាស់មនុស្ស។ លក្ខណៈពិសេសរបស់វាជាការចូលរួមនៃម៉ូដែល generative AI ដែលអាចសម្តែងសន្ទនាដូចជា​មនុស្ស​ដោយមានភាពរំភើប និងបង្កើតចម្លើយផ្អែកលើចម្ងល់និងបរិបទ។ កម្មវិធីជជែកដែលបានបង្កើតដោយ generative AI អាចមានការពិភាក្សាប្រភេទបើកចំហ មានការចេញបរិបទដែលបត់បែន និងផលិតសន្ទនាដែលមានការច្នៃប្រឌិតឬស្មុគស្មាញ។

- **ប្រសិទ្ធភាពល្អជាង**៖ នៅពេលសាងសង់មុខងារពីដើម អ្នកនឹងសួរថា "តើវាអាចធ្វើការកំណត់បរិមាណបានយ៉ាងដូចម្តេច? តើកម្មវិធីនេះអាចទទួលអ្នកប្រើថ្មីច្រើនទេ?" SDK និង API ធ្វើការអភិវឌ្ឍយ៉ាងល្អភាគច្រើនមានដំណោះស្រាយរួមបញ្ចូល។
- **ការថែទាំងាយស្រួល**៖ ការអាប់ដែតនិងការកែលម្អងាយស្រួល ដោយសារតែភាគច្រើន API និង SDK ត្រូវការរត់ការអាប់ដែតដោយផ្នែកបណ្ណាល័យនៅពេលមានជំនាន់ថ្មីត្រូវចេញផ្សាយ។
- **ចូលដល់បច្ចេកវិទ្យាថ្មីៗ**៖ ការប្រើម៉ូដែលដែលបានកែតម្រូវ និងហ្វឹកហាត់លើទិន្នន័យធំនាំឲ្យកម្មវិធីរបស់អ្នកមានសមត្ថភាពភាសាធម្មជាតិ។

ការចូលដំណើរការមុខងាររបស់ SDK ឬ API ប្រហែលជាត្រូវការការអនុញ្ញាតឲ្យប្រើសេវាកម្មដែលផ្តល់ជូន ខ្លឹមសារញឹកញាប់តាមរយៈកូនសោគុណឬសញ្ញាសម្គាល់ស្នាក់នៅ។ យើងនឹងប្រើប្រាស់បណ្ណាល័យ Python របស់ OpenAI ដើម្បីสำรวจលក្ខណៈនេះ។ អ្នកក៏អាចសាកល្បងផ្ទាល់នៅក្នុង [សៀវភៅកំណត់ហេតុសម្រាប់ OpenAI](./python/oai-assignment.ipynb?WT.mc_id=academic-105485-koreyst) ឬ [សៀវភៅកំណត់ហេតុសម្រាប់ Azure OpenAI Services](./python/aoai-assignment.ipynb?WT.mc_id=academic-105485-koreys) សម្រាប់មេរៀននេះ។

from openai import OpenAI

## បទពិសោថ្នរ (UX)

គោលការណ៍ UX ទូទៅអាចប្រើសម្រាប់កម្មវិធីជជែក ប៉ុន្តែមានការពិចារណាពិសេសមួយចំនួនដែលមានសារៈសំខាន់ដោយសារតែកម្មវិធីរៀនម៉ាស៊ីន។

- **មេកានិកសម្រាប់ដោះស្រាយភាពចម្ងល់**៖ ម៉ូដែល generative AI មួយចំនួនឆ្លើយពិចារណាជាការអាប់អរ។ មុខងារដែលអនុញ្ញាតឲ្យអ្នកប្រើសួរពន្យល់បានអាចជួយបានបើពួកគេប្រទះរឿងនេះ។
- **ការចងចាំបរិបទ**៖ ម៉ូដែល generative AI កំរិតខ្ពស់អាចចងចាំបរិបទក្នុងការសន្ទនា ដែលជាគ្រឿងបន្លាស់ដែលចាំបាច់។ ការផ្តល់ឲ្យអ្នកប្រើមានសិទ្ធិក្នុងការគ្រប់គ្រងនិងគ្រប់គ្រងបរិបទធ្វើឲ្យបទពិសោធន៍ប្រើល្អប្រសើរឡើង ប៉ុន្តែបណ្តាលឲ្យមានហានិភ័យនៃការរក្សាទុកព័ត៌មានសំងាត់របស់អ្នកប្រើ។ ការពិចារណាពីរយៈពេលរក្សាទុក នឹងដូចជាការណែនាំគោលនយោបាយរក្សាទុកអាចតុល្យភាពរវាងតម្រូវការបរិបទ និងភាពឯកជន។
- **ការផ្ទាល់ខ្លួន**៖ ជាមួយនឹងសមត្ថភាពរៀននិងបត់បែន ម៉ូដែល AI ផ្តល់បទពិសោធន៍ឯកជនទៅអ្នកប្រើ។ ការរៀបចំបទពិសោធន៍តាមរយៈមុខងារ ដូចជា “ប្រវត្តិរូបអ្នកប្រើ” មិនត្រឹមតែធ្វើឲ្យអ្នកប្រើមានអារម្មណ៍ថាត្រូវបានយល់ដឹង ប៉ុន្តែបង្រៀនខួរក្បាលរបស់ពួកគេក្នុងការស្វែងរកចម្លើយជាក់លាក់ ដែលបង្កើតជាការជជែកមានប្រសិទ្ធភាពនិងពេញចិត្ត។

### សមត្ថភាពចូលដូចជា Accessibility

មិនថាអ្នកប្រើមានខូចខាតការមើល ការស្តាប់ រឺចលនា ឬការយល់ដឹង កម្មវិធីជជែកដែលត្រូវបានរចនាដោយល្អគួរតែអាចប្រើបានសម្រាប់មនុស្សគ្រប់គ្នា។ តារាងខាងក្រោមបង្ហាញពីមុខងារជាក់លាក់សម្រាប់ជួយឱ្យមានសមត្ថភាពចូលដល់សម្រាប់មនុស្សដែលមានការ​ខូចខាតហ្នឹងៗ។

- **មុខងារសម្រាប់ការខូចខាតមើល**៖ ទម្រង់ខុសគ្នាខ្ពស់ និងអក្សរប្ដូរទំហំបាន សមត្ថភាពជាមួយកម្មវិធីអានអេក្រង់។
- **មុខងារសម្រាប់ការខូចខាតស្តាប់**៖ មុខងារ​អក្សរទៅសំលេង និងសំលេងទៅអក្សរ រួមជាមួយសញ្ញាអូឌីយ៉ូ។
- **មុខងារសម្រាប់ការខូចខាតចលនា**៖ គាំទ្រជាមួយក្តារចុច និងបញ្ជារដោយសម្លេង។
- **មុខងារសម្រាប់ការខូចខាតផ្នែកផ្លូវការយល់ដឹង**៖ ជម្រើសភាសាលម្អិតងាយស្រួល។

សូមនឹកឃើញកម្មវិធីជជែកមួយដែលយល់ពាក្យបច្ចេកទេសរបស់ក្រុមហ៊ុនអ្នក និងប្រាកដថាសំណួរដែលអ្នកប្រើភាគច្រើនមានត្រូវបានចំអិន។ មានរបៀបខ្លះគួររៀបរាប់៖

- **ប្រើប្រាស់ម៉ូដែល DSL**. DSL មានន័យថា domain specific language។ អ្នកអាចប្រើម៉ូដែល DSL ដែលបានហ្វឹកហាត់ផ្ទាល់លើដែនជាក់លាក់ដើម្បីយល់ពីកូដនិងសេនារីយ៉ូ។
- **អនុវត្ត fine-tuning**. Fine-tuning គឺជាការបន្តហ្វឹកហាត់ម៉ូដែលអ្នកជាមួយទិន្នន័យជាក់លាក់។

## ការប្ដូរតម្រូវ៖ ប្រើ DSL

ការប្រើម៉ូដែលភាសា domain-specific (DSL Models) អាចបង្កើនការចូលរួមរបស់អ្នកប្រើដោយផ្តល់ការប្រាស្រ័យទាក់ទងពិសេស ដែលអាចទាក់ទងជាមួយបរិបទ។ វាជាប្រភេទម៉ូដែលដែលបានហ្វឹកហាត់ ឬបាន fine-tune ដើម្បីយល់ និងបង្កើតអត្ថបទពាក់ព័ន្ធនឹងវិស័យ ឧស្សាហកម្ម ឬប្រធានបទជាក់លាក់។ ជម្រើសប្រើម៉ូដែល DSL អាចតំឡើងពី អនុវត្តហ្វឹកហាត់ពីដើម មករហូតដល់ប្រើវាតាម SDK និង API ដែលមានស្រាប់។ ជម្រើសមួយទៀតគឺ fine-tuning ដែលរួមបញ្ចូលការយកម៉ូដែលដែលបានហ្វឹកហាត់រួចហើយ និងកែប្រែលើដែនជាក់លាក់។

## ការប្ដូរតម្រូវ៖ អនុវត្ត fine-tuning

Fine-tuning ត្រូវបានគេចាត់ទុកពេលម៉ូដែលដែលបានហ្វឹកហាត់រួចមិនគ្រប់គ្រាន់សម្រាប់ដែនពិសេសឬបេសកកម្មជាក់លាក់។

ឧទាហរណ៍ សំណួរពេទ្យមានភាពស្មុគស្មាញ និងត្រូវការបរិបទជាច្រើន។ នៅពេលអ្នកជំនាញពេទ្យវិនិច្ឆ័យមនុស្សព្យាបាល គឺដោយផ្អែកលើហេតុផលច្រើនដូចជារបៀបរស់នៅ ឬជំងឺដែលមានមកពីមុន ហើយអាចប្រើការស្រាវជ្រាវពេទ្យថ្មីៗ ដើម្បីផ្តល់ការវិនិច្ឆ័យត្រឹមត្រូវ។ ក្នុងសំណាក់ករណីថ្នាក់នេះ កម្មវិធីជជែក AI ដែលមានគោលបំណងទូទៅមិនអាចជាវត្ថុពិតប្រាកដបាន។

### សេណារីយៈកម្មវិធីពេទ្យ

សូមគិតដល់កម្មវិធីជជែកមួយដែលរចនាឡើងសម្រាប់ជួយអ្នកវិជ្ជាជីវៈពេទ្យនៅក្នុងការផ្តល់ជូនការពិគ្រោះផ្លូវការព្យាបាល, ប្រតិបត្តិការស្ថិតិនៃថ្នាំ ឬការស្រាវជ្រាវថ្មីៗ។

ម៉ូដែលទូទៅអាចគ្របដណ្តប់ការឆ្លើយសំណួរពេទ្យមូលដ្ឋាន ឬផ្តល់ព្រឹតិកម្មទូទៅ ប៉ុន្តេច្រើនពេលវាចាញ់ចំពោះ៖

- **ករណីពិសេសស្មុគស្មាញខ្ពស់**៖ ឧទាហរណ៍វេជ្ជបណ្ឌិតអាចសួរថា "តើអ្វីទៅជាប្រព័ន្ធការគ្រប់គ្រងជំងេត្ត្រឡាប់ថ្នាំតំបន់ប្រសិទ្ធិភាពខ្ពស់សម្រាប់អ្នកជំងឺក្មេង?"
- **ខ្វះភាពទាន់សម័យ**៖ ម៉ូដែលទូទៅអាចបរាជ័យក្នុងការផ្តល់ចម្លើយទាន់សម័យដែលរួមបញ្ចូលការអភិវឌ្ឍចុងក្រោយនៅវិស័យប្រាក់ថ្នាំនិងប្រព័ន្ធប្រសាទ។

ក្នុងករណីបែបនេះ ការថែរក្សាម៉ូដែលជាមួយទិន្នន័យពេទ្យជាក់លាក់អាចកែលម្អសមត្ថភាពប្រសិទ្ធភាពនៃការឆ្លើយសំណួរដល់ត្រឹមត្រូវ។ វាត្រូវការព័ត៌មានទិន្នន័យធំបំផុតនិងពាក់ព័ន្ធដែលតំណាងឲ្យបញ្ហា និងសំណួរសម្រាប់ដែនជាក់លាក់។
