# មាតិកាទីមុខអំពី Generative AI និង ម៉ូដែលភាសាធំបំផុត

Generative AI គឺជាបញ្ញាសិប្បនិម្មិតដែលមានសមត្ថភាពបង្កើតអត្ថបទ រូបភាព និងមាតិកាប្រភេទផ្សេងៗទៀត។ អ្វីដែលធ្វើអោយវាជាបច្ចេកវិទ្យាដែលអស្ចារ្យគឺវាធ្វើអោយ AI គ្រប់គ្នាអាចប្រើបាន ដោយគ្រាន់តែមានពាក្យបង្ហាញអត្ថបទ ឬប្រយោគមួយសរសេរជាភាសាប្រែកម្មវិធីធម្មជាតិ។ មិនចាំបាច់អោយអ្នករៀនភាសាដូចជា Java ឬ SQL ដើម្បីធ្វើអ្វីដែលមានតម្លៃទេ គ្រាន់តែអ្នកប្រើភាសារបស់អ្នក អ្នកប្រាប់អ្វីដែលអ្នកចង់ និង AI នឹងផ្ដល់អនុសាសន៍មកវិញពីម៉ូដែល AI។ ការដាក់ពាក្យប្រើប្រាស់ និងផលប៉ះពាល់សម្រាប់នេះមានទំហំធំ អ្នកអាចសរសេរឬយល់ពីរបាយការណ៍ សរសេរកម្មវិធី និងច្រើនទៀត សុទ្ធតែក្នុងរយៈពេលប៉ុន្មានវិនាទី។

ក្នុងមេរៀននេះ យើងនឹងស្វែងយល់ពីរបៀបដែលក្រុមហ៊ុនបង្កើតថ្មីរបស់យើងប្រើប្រាស់ generative AI ដើម្បីបើកសេនារីយ៉ូថ្មីៗក្នុងពិភពការអប់រំ និងរបៀបដែលយើងដោះស្រាយបញ្ហាដែលមិនអាចជៀសវាងបានពាក់ព័ន្ធនឹងផលប៉ះពាល់សង្គមនៃការត្រូវបានអនុវត្ត និងការបង្ខំចិត្តបច្ចេកវិទ្យា។

## មាតិកាទីមុខ

មេរៀននេះនឹងជួយលោកអ្នកយល់ពី៖

- មាតិកាទីមុខនៃសេនារីយ៉ូអាជីវកម្ម៖ គំនិតនិងបេសកកម្មរបស់ក្រុមហ៊ុនបង្កើតថ្មីរបស់យើង។
- Generative AI និងរបៀបដែលយើងបានឈានដល់ទេសភាពបច្ចេកវិទ្យាបច្ចុប្បន្ន។
- របៀបដំណើរការខាងក្នុងរបស់ម៉ូដែលភាសាធំបំផុត។
- សមត្ថភាពសំខាន់ៗ និងករណីប្រើប្រាស់ជាក់ស្តែងនៃម៉ូដែលភាសាធំបំផុត។

## គោលបំណងការសិក្សា

បន្ទាប់ពីបញ្ចប់មេរៀននេះ អ្នកនឹងយល់ពី៖

- តើ generative AI ជាអ្វី និងរបៀបដំណើរការរបស់ម៉ូដែលភាសាធំបំផុត។
- របៀបដែលអ្នកអាចប្រើប្រាស់ម៉ូដែលភាសាធំបំផុតសម្រាប់ករណីប្រើប្រាស់ផ្សេងៗ ដោយផ្ដោតលើសេនារីយ៉ូអប់រំ។

## សេនារីយ៉ូ៖ ក្រុមហ៊ុនបង្កើតថ្មីផ្នែកអប់រំរបស់យើង

បញ្ញាសិប្បនិម្មិតជំនាន់ថ្មី (Generative Artificial Intelligence - AI) តំណាងឱ្យកំពូលនៃបច្ចេកវិទ្យា AI ដែលបានធ្វើឱ្យអាចធ្វើអ្វីដែលមិនគួរឱ្យជឿពីមុន។ ម៉ូដែល generative AI មានសមត្ថភាពនិងកម្មវិធីជាច្រើន ប៉ុន្តែក្នុងមេរៀននេះ យើងនឹងស្វែងរករបៀបនៃការប្ដូរជំនាន់ប្រាជ្ញានៅក្នុងវិស័យអប់រំតាមរយៈក្រុមហ៊ុនបង្កើតថ្មីមួយដែលមានរឿងរ៉ាវ។ យើងនឹងហៅក្រុមហ៊ុននេះថា _ក្រុមហ៊ុនបង្កើតថ្មីរបស់យើង_។ ក្រុមហ៊ុនរបស់យើងកំពុងធ្វើការក្នុងវិស័យអប់រំបែបមានបេសកកម្មមួយដែលមានភាពអំណានថា

> _បង្កើនការចូលដំណើរការក្នុងការសិក្សា នៅលើកម្រិតពិភពលោក ដើម្បីធ្វើឱ្យមានការចូលដំណើរការការអប់រំយ៉ាងតុល្យភាព និងផ្តល់បទពិសោធន៍សិក្សាផ្ទាល់ខ្លួនទៅនិស្សិតនីមួយៗ ដោយផ្អែកលើតម្រូវការរបស់ពួកគេ_។

ក្រុមការងារបង្កើតថ្មីរបស់យើងដឹងថា យើងមិនអាចសម្រេចបានគោលបំណងនេះដោយមិនប្រើប្រាស់ឧបករណ៍ដ៏មានអំណាចមួយ – ម៉ូដែលភាសាធំបំផុត (LLMs)។

Generative AI ត្រូវបានរំពឹងទុកថានឹងបំផ្លាញរបៀបដែលយើងរៀននិងបង្រៀននៅថ្ងៃនេះ ជាមួយនិស្សិតដែលមានគ្រូវីរុល 24 ម៉ោងក្នុងមួយថ្ងៃផ្ដល់ព័ត៌មាន និងឧទាហរណ៍ច្រើនទៀត ហើយគ្រូបង្រៀនអាចប្រើឧបករណ៍ច្នៃប្រឌិតដើម្បីវាយតម្លៃនិស្សិតរបស់ពួកគេនិងផ្តល់មតិយោបល់។

## តើតើយើងបាន Generative AI ដោយរបៀបណា?

ទោះបីជាការផ្សព្វផ្សាយដ៏ប៉ិនប្រសប់បច្ចុប្បន្នមានចំពោះម៉ូដែល generative AI ក៏ដោយ បច្ចេកវិទ្យានេះកំពុងត្រូវបានអភិវឌ្ឍនាដល់ជាច្រើនទសវត្សរ៍មកហើយ ជាមួយនឹងការស្រាវជ្រាវដំបូងដែលចាប់ផ្ដើមតាំងពីទសវត្សរ៍ 60។ ឥឡូវនេះយើងស្ថិតនៅចំណុចដែល AI មានសមត្ថភាពច្នៃប្រឌិតដូចមនុស្ស ដូចជាការសន្ទនា ដែលបង្ហាញឲ្យឃើញពីប៉ុន្មានវត្តមានដូចជា [OpenAI ChatGPT](https://openai.com/chatgpt) ឬ [Microsoft Copilot](https://copilot.microsoft.com/?WT.mc_id=academic-105485-koreyst) ដែលក៏ប្រើម៉ូដែល GPT សម្រាប់បទពិសោធន៍ស្វែងរកតាមបណ្តាញ។

វិលត្រឡប់ទៅពីនេះពេលខ្លី ប្រភេទម៉ូដែល AI ដំបូងៗ គឺជាចតុមុខតែនៈ បញ្ជូលជជែកជាមួយវា ប្រើប្រាស់មូលដ្ឋានចំណេះដឹងដែលបានយកចេញពីក្រុមអ្នកជំនាញ ហើយតំណាងឲ្យក្នុងកុំព្យូទ័រ។ ចម្លើយក្នុងមូលដ្ឋានចំណេះដឹងត្រូវបានចលនា ដោយពាក្យគន្លឹះបង្ហាញក្នុងអត្ថបទបញ្ចូល។
ទោះជាយ៉ាងណា វារួចមកថា វិធីនេះមិនអាចបង្រ្កាបបានល្អនោះឡើយ។

### វិធីសាស្ត្រជា ស្ថិតិមួយចំពោះ AI៖ ការសិក្សាម៉ាស៊ីន

ចំណុចបំលាស់បំរួលបានមក នៅទសវត្សរ៍ 90 ជាមួយនឹងការដាក់បញ្ចូលវិធីសាស្ត្រស្ថិតិទៅក្នុងការវិភាគអត្ថបទ។ នេះបាននាំឲ្យមានការអភិវឌ្ឍនាឡើងវិញនូវគន្លងថ្មីៗ – ហៅថាការសិក្សាម៉ាស៊ីន – ដែលមានសមត្ថភាពរៀនទម្រង់ពីទិន្នន័យដោយគ្មានការបញ្ចូលកម្មវិធីច្បាស់លាស់។ វិធីសាស្ត្រនេះអនុញ្ញាតឲ្យឧបករណ៍ធ្វើតួដូចការយល់ដឹងភាសាមនុស្ស៖ ម៉ូដែលស្ថិតិត្រូវបានបណ្តុះលើមាតិការអត្ថបទនិងស្លាក ដែលជួយឲ្យម៉ូដែលចាត់ថ្នាក់អត្ថបទបញ្ចូលមិនស្គាល់ដោយស្លាកដែលបង្ហាញបំណងនៃសារនោះ។

### បណ្តាញសរសៃប្រសាទ និងអ្នកជំនួយវីរុលសម័យទំនើប

នៅប៉ុន្មានឆ្នាំថ្មីៗនេះ ការវិវឌ្ឍបច្ចេកវិទ្យាដែកឧបករណ៍ ដែលអាចដោះស្រាយទិន្នន័យធំៗ និងប្រតិបត្តិគណិតកិច្ចស្មុគស្មាញ បានបណ្តល់កម្លាំងស្រាវជ្រាវក្នុង AI ដែលបាននាំទៅរកអភិវឌ្ឍន៍គន្លងការសិក្សាម៉ាស៊ីនដែលមានកម្រិតខ្ពស់ហៅថាបណ្តាញសរសៃប្រសាទ ឬក៏គណិតវិទ្យាដែលហៅថា deep learning។

បណ្តាញសរសៃប្រសាទ (ជាពិសេស Recurrent Neural Networks – RNNs) បានបង្កើនភាពខ្ពស់ទៅលើកំណត់ធម្មជាតិភាសា ដោយអនុញ្ញាតឲ្យតំណាងន័យនៃអត្ថបទក្នុងរបៀបមានន័យជាងមុន សំរាប់តម្លៃតំបន់នៃពាក្យមួយនៅក្នុងប្រយោគ។

នេះជាបច្ចេកវិទ្យាដែលជំរុញអ្នកជំនួយវីរុលដែលកើតមានឡើងនៅទសវត្សរ៍ដំបូងនៃសតវត្សរ៍ថ្មី ដោយមានជំនាញខ្ពស់ក្នុងការបកប្រែភាសាមនុស្ស កំណត់តម្រូវការ និងអនុវត្តអំពើតាមដើម្បីបំពេញ តទៅដូចជាការឆ្លើយតបជាមួយស្ដេក។​

### សព្វថ្ងៃ Generative AI

នេះហើយជារបៀបដែលយើងបានឈានទៅ Generative AI ដែលអាចចាត់ទុកថាជាផ្នែកមួយនៃ deep learning។

![AI, ML, DL និង Generative AI](../../../translated_images/km/AI-diagram.c391fa518451a40d.webp)

បន្ទាប់ពីការស្រាវជ្រាវយូរឆ្នាំក្នុងវិស័យ AI ស្ថាបត្យកម្មម៉ូដែលថ្មីមួយ – ហៅថា _Transformer_ – បានលើកកម្រិតកំណត់ឆ្ងាយរបស់ RNNs ដោយមានសមត្ថភាពទទួលខ្សែអក្សរយ៉ាងវែងជាងមុន។ Transformers មានមូលដ្ឋានលើយន្តការប្រុងប្រយ័ត្ន ដែលអនុញ្ញាតឲ្យម៉ូដែលផ្ដល់ទម្ងន់ខុសគ្នាទៅនឹងបញ្ចូលដែលវារកបាន ដោយ ‘យកចិត្តទុកដាក់ជាចម្បង’ ទីកន្លែងដែលមានព័ត៌មានសំខាន់បំផុត ឥតគិតថាតំរៀបវាចេញ។

ម៉ូដែល generative AI ចុងក្រោយៗភាគច្រើន – ហៅថា ម៉ូដែលភាសាធំបំផុត (LLMs) ព្រោះពួកវាសម្របសម្រួលជាមួយអត្ថបទបញ្ចូល និងចាកចេញ – នៅពិតជា​មានមូលដ្ឋានលើស្ថាបត្យកម្មនេះ។ អ្វីដែលគួរឱ្យចាប់អារម្មណ៍ប្រសើរពីម៉ូដែលទាំងនេះ រួម​មាន ការបណ្តុះបណ្តាលលើទិន្នន័យដែលមិនមានស្លាកពីប្រភពផ្សេងៗដូចជា សៀវភៅ អត្ថបទ និងគេហទំព័រ – គឺពួកវាអាចសម្របខ្លួនចំពោះភារកិច្ចជាច្រើន ហើយបង្កើតអត្ថបទត្រឹមត្រូវគ្រប់វេយ្យាករណ៍ដែលមានសញ្ញានៃច្នៃប្រឌិត។ ដូច្នេះ មិនត្រឹមតែពួកវាបង្កើនសមត្ថភាពនៃម៉ាស៊ីនក្នុងការយល់អត្ថបទបញ្ចូលទេ ប៉ុន្តែវាបង្កើតសមត្ថភាពបង្កើតចម្លើយធម្មតាជាភាសាថ្មីបាន។

## ម៉ូដែលភាសាធំបំផុតដំណើរការបែបណា?

នៅជំពូកបន្ទាប់យើងនឹងស្វែងយល់អំពីម៉ូដែល Generative AI ប្រភេទផ្សេងៗ ប៉ុន្តែសម្រាប់ឥឡូវនេះតោះមើលរបៀបដែលម៉ូឌែលភាសាធំបំផុតដំណើរការ ជាមួយការផ្ដោតលើម៉ូឌែល OpenAI GPT (Generative Pre-trained Transformer)។

- **Tokenizer, អត្ថបទទៅជាលេខ**៖ ម៉ូឌែលភាសាធំបំផុតទទួលបានអត្ថបទជាបញ្ចូលនិងបង្កើតអត្ថបទជាចំណែកចេញ។ ទោះយ៉ាងណា ជា៉ម៉ូដែលស្ថិតិ វាដំណើរការល្អជាងជាមួយលេខជាងអត្ថបទ។ ដូច្នេះ មួយនៅក្នុងបញ្ចូលទៅម៉ូដែលត្រូវបានកំណត់ដោយ tokenizer មុនពេលប្រើដោយម៉ូដែលគោល។ Token គឺជាផ្នែកមួយនៃអត្ថបទ – មានចំនួនតួអក្សរមិនថេរ ដូច្នេះភារកិច្ចសំខាន់សម្រាប់ tokenizer គឺបំបែកបញ្ចូលទៅជា token ជាច្រើន។ បន្ទាប់មក Token ត្រូវបានផ្គូរផ្គងជាមួយលេខសម្គាល់ token ដែលជាការរចនារចនាសម្ព័ន្ធលេខគត់នៃផ្នែកអត្ថបទដើម។

![ឧទាហរណ៍នៃ tokenization](../../../translated_images/km/tokenizer-example.80a5c151ee7d1bd4.webp)

- **ការព្យាករណ៍ token ចេញ**៖ ពី token n នៅជាបញ្ចូល (ដែលចំនួនអតិបរមាអាចខុសគ្នាចាប់ពីម៉ូដែលមួយទៅម៉ូដែលផ្សេង) ម៉ូដែលអាចព្យាករណ៍ token មួយជា​ចេញ។ Token នេះបន្ទាប់មកត្រូវបញ្ចូលក្នុងបញ្ចូលនៃតំណាក់កាលបន្ទាប់ក្នុងរបៀបបង្ហះជាបង្អួច ដែលអនុញ្ញាតឲ្យប្រើប្រាស់ប្រាកដវិញនៅក្នុងការទទួលបានអ្នកប្រើសំណួរមួយឬច្រើនជាប្រយោគឆ្លើយតប។ នេះជាការស្រាយពីហេតុផលដែល បើអ្នកធ្លាប់លេងជាមួយ ChatGPT អ្នកអាចចាប់អារម្មណ៍ថា វាដូចជាស្ទាក់នៅកណ្តាលប្រយោគណាមួយ។

- **ដំណើរការជ្រើសរើស, គណនាភាពប្រហែល**៖ token ចេញត្រូវបានជ្រើសរើសដោយម៉ូដែលដោយយោងទៅលើការគណនាភាពប្រាកដនៃការកើតមានបន្ទាប់ពីខ្សែអក្សរបច្ចុប្បន្ន។ ទាំងនេះដោយសារ ម៉ូដែលព្យាករណ៍ចែកចាយភាពប្រហែលលើ​‘token បន្ទាប់ទាំងអស់’ ដែលគណនាដោយផ្អែកលើការបណ្តុះបណ្តាលរបស់វា។ ទោះជាយ៉ាងណា មិនមែនtoken ដែលមានភាពប្រហែលខ្ពស់ជាងគេត្រូវបានជ្រើសរើសពីចែកចាយនោះនោះទេ។ មានកម្រិតមួយនៃភាពចៃដន្យបន្ថែមដើម្បីឲ្យម៉ូដែលដំណើរការជាវិធីមិនកំណត់ - យើងមិនទទួលបានចម្លើយដដែលគ្នាសម្រាប់បញ្ចូលដដែលទេ។ កម្រិតនៃភាពចៃដន្យនេះត្រូវបានបន្ថែមដើម្បីច្នៃប្រឌិតគិតគូរនិងវាអាចត្រូវបានកំណត់តាមតម្លៃម៉ូដែលមួយដែលហៅថា temperature។

## តើយើងអាចប្រើម៉ូដែលភាសាធំបំផុតយ៉ាងដូចម្តេច?

ឥឡូវនេះយើងមានយល់ដឹងល្អពីរបៀបដំណើរការខាងក្នុងរបស់ម៉ូឌែលភាសាធំបំផុត យើងមកមើលឧទាហរណ៍ជាក់ស្តែងមួយចំនួននៃភារកិច្ចដែលពួកវាធ្វើបានល្អ ជាមួយការផ្ដោតលើសេនារីយ៉ូអាជីវកម្មរបស់យើង។
យើងបាននិយាយថា សមត្ថភាពសំខាន់នៃម៉ូឌែលភាសាធំបំផុត គឺ _បង្កើតអត្ថបទពីចំណុចដើម ចាប់ផ្ដើមពីអត្ថបទបញ្ចូលដែលសរសេរជាភាសាធម្មជាតិ_។

- ផ្នែកនៃ **អត្ថបទដែលត្រូវបញ្ចប់** ដែលក្នុងន័យគឺជាការស្នើសុំជំនួយសរសេរ។
  
  ![ឧទាហរណ៍ការបញ្ចប់អត្ថបទ](../../../translated_images/km/text-completion-example.cbb0f28403d42752.webp)

- ផ្នែកនៃ **កូដ** ជាមួយនឹងសំណើរ សូមជំរុញ និងឯកសារពិពណ៌នា ឬមតិយោបល់ស្នើបង្កើតកូដមួយសំរាប់ភារកិច្ចជាក់លាក់។

ឧទាហរណ៍ខាងលើគឺសាមញ្ញ និងមិនមានគោលបំណងបង្ហាញគ្រប់សមត្ថភាពនៃម៉ូដែលភាសាធំបំផុតទាំងអស់ទេ។ វាមានគោលបំណងបង្ហាញពីសក្តានុពលនៃការប្រើប្រាស់ generative AI ជាពិសេសក្នុងបរិបទអប់រំ។

ក៏ដូចជាចេញពីម៉ូដែល generative AI មិនពេញលេញ និងពេលខ្លះភាពច្នៃប្រឌិតរបស់ម៉ូដែលអាចបង្ករឲ្យមានចេញមក ជាការប្រមូលបញ្ចូលពាក្យដែលអ្នកប្រើប្រាស់មនុស្សអាចយល់ថាជាការលាក់បោកប្រាស់ពិភពលោក ឬអាចមានការប្រហែលពាក្យមិនល្អ។ Generative AI មិនមែនជាជំនាញផ្លូវចិត្តក៏ទេ៖ យ៉ាងហោចណាស់ ក្នុងនិយមន័យទូលំទូលាយ រួមមានការគិតវិនិច្ឆ័យ និងច្នៃប្រឌិត គ្រប់ផ្នែកនៃចិត្តវិទ្យា; វាមិនមែនជាវិធីដឹងច្បាស់ទេ ហើយវាមិនគួរឱ្យទុកចិត្តទេ ពីព្រោះការបង្កើតមិនត្រឹមត្រូវដូចជាការយោងខុស, មាតិកា និងពាក្យប្រកាស អាចត្រូវបានរួមបញ្ចូលជាមួយព័ត៌មានត្រឹមត្រូវ និងបង្ហាញដោយប៉ះពាល់ និងទ្រង់ទ្រាយទាំងមូល។ ក្នុងមេរៀនបន្ទាប់ យើងនឹងពិនិត្យប្រឹងប្រែងមកវិញចំពោះការដំណើរការទាំងនេះនឹងមើលថាតើយើងអាចធ្វើអ្វីដើម្បីកាត់បន្ថយវា។

## ហត្ថការងារ

ការងាររបស់អ្នកគឺរសៀលស្រាវជ្រាវបន្ថែមអំពី [generative AI](https://en.wikipedia.org/wiki/Generative_artificial_intelligence?WT.mc_id=academic-105485-koreyst) ហើយព្យាយាមស្វែងរកតំបន់ណាមួយដែលអ្នកនឹងបន្ថែម generative AI នៅថ្ងៃនេះដែលមិនមានវា។ តើផលប៉ះពាល់នឹងផ្លាស់ប្តូរយ៉ាងដូចម្តេចពីការធ្វើវាជា "វិធីចាស់" អ្នកអាចធ្វើអ្វីមួយដែលមុនមិនអាចធ្វើបានរឺថាអ្នកមានល្បឿនលឿនជាងមុន? សរសេរសង្ខេប 300 ពាក្យពីក្រុមហ៊ុនសុបិនប្រាជ្ញា AI របស់អ្នក និងរួមបញ្ចូលចំណងជើងដូចជា "បញ្ហា", "របៀបដែលខ្ញុំប្រើ AI", "ផលប៉ះពាល់" និងជាជម្រើសផែនការអាជីវកម្ម។
