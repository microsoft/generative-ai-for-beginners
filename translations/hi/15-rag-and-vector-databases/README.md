    words = text.split()
    chunks = []
    current_chunk = []

        chunks.append(' '.join(current_chunk))

user_input = "what is a perceptron?"

def chatbot(user_input):
    # рдкреНрд░рд╢реНрди рдХреЛ рдХреНрд╡реЗрд░реА рд╡реЗрдХреНрдЯрд░ рдореЗрдВ рдмрджрд▓реЗрдВ
