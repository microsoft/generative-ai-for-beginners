```python
    for i in range(num_iterations):
        pos = random.choice(positive_examples)
        neg = random.choice(negative_examples)

        z = np.dot(pos, weights) # ‡∞™‡∞∞‡±ç‡∞∏‡±Ü‡∞™‡±ç‡∞ü‡±ç‡∞∞‡∞æ‡∞®‡±ç ‡∞Ö‡∞µ‡±Å‡∞ü‡±ç‡∞™‡±Å‡∞ü‡±ç‚Äå‡∞®‡±Å ‡∞≤‡±Ü‡∞ï‡±ç‡∞ï‡∞ø‡∞Ç‡∞ö‡∞Ç‡∞°‡∞ø
        if z < 0: # ‡∞∏‡∞æ‡∞®‡±Å‡∞ï‡±Ç‡∞≤ ‡∞â‡∞¶‡∞æ‡∞π‡∞∞‡∞£‡∞®‡±Å ‡∞™‡±ç‡∞∞‡∞§‡∞ø‡∞ï‡±Ç‡∞≤‡∞Ç‡∞ó‡∞æ ‡∞µ‡∞∞‡±ç‡∞ó‡±Ä‡∞ï‡∞∞‡∞ø‡∞Ç‡∞ö‡∞æ‡∞∞‡±Å
            weights = weights + eta*weights.shape

        z  = np.dot(neg, weights)
        if z >= 0: # ‡∞™‡±ç‡∞∞‡∞§‡∞ø‡∞ï‡±Ç‡∞≤ ‡∞â‡∞¶‡∞æ‡∞π‡∞∞‡∞£‡∞®‡±Å ‡∞∏‡∞æ‡∞®‡±Å‡∞ï‡±Ç‡∞≤‡∞Ç‡∞ó‡∞æ ‡∞µ‡∞∞‡±ç‡∞ó‡±Ä‡∞ï‡∞∞‡∞ø‡∞Ç‡∞ö‡∞æ‡∞∞‡±Å
            weights = weights - eta*weights.shape

    return weights
```
