    words = text.split()
    chunks = []
    current_chunk = []

    for word in words:
        current_chunk.append(word)
        if len(' '.join(current_chunk)) < max_length and len(' '.join(current_chunk)) > min_length:
            chunks.append(' '.join(current_chunk))
            current_chunk = []

    # ‡∞ö‡∞ø‡∞µ‡∞∞‡∞ø ‡∞≠‡∞æ‡∞ó‡∞Ç ‡∞ï‡∞®‡±Ä‡∞∏ ‡∞™‡±ä‡∞°‡∞µ‡±Å ‡∞ö‡±á‡∞∞‡∞ï‡∞™‡±ã‡∞§‡±á ‡∞ï‡±Ç‡∞°‡∞æ ‡∞¶‡∞æ‡∞®‡±ç‡∞®‡∞ø ‡∞ú‡±ã‡∞°‡∞ø‡∞Ç‡∞ö‡∞Ç‡∞°‡∞ø
    if current_chunk:
        chunks.append(' '.join(current_chunk))
